74  // Get state of vehicle that we are trailing

76  for (auto it = contacts_->begin(); it != contacts_->end(); it++) {
77  if (trail_id_ == it->second.id().id()) {
78  trail_state = it->second.state();
79  break;
80  }
81  }
82 
83  // Compute the direction of the trail point relative to the other entity's
84  // center. Rotate a unit vector pointing towards the x-axis around the
85  // z-axis by the trail angle. Then rotate that vector by the other entity's
86  // orientation.
87  Eigen::Vector3d direction = trail_state->quat() * aa_angle_elev_ *
88  aa_angle_az_ * Eigen::Vector3d::UnitX();
89 
90  // Compute the position of the desired trail point
91  Eigen::Vector3d trail_point = trail_state->pos() + direction * trail_range_;
92 
93  // Compute the positio of the desisired trail point, taking the other
94  // entity's velocity into consideration
95  Eigen::Vector3d trail_point_w_vel = trail_point + trail_state->vel();
96 
97  // Compute the vector pointing from our position to the trail_point_w_vel
98  desired_vector_ = trail_point_w_vel - state_->pos();
99 
100  // Normalize the vector to the maximum length
101  if (desired_vector_.norm() > max_vector_length_) {
102  desired_vector_ = desired_vector_.normalized() * max_vector_length_;
103  }
104 
105  // Decrease speed as we get closer to the actual track point
106  double trail_point_diff = (state_->pos() - trail_point).norm();
107  if (trail_point_diff < desired_vector_.norm()) {
108  double speed = desired_vector_.norm() - (state_->vel().norm() - trail_state->vel().norm());
109  desired_vector_ = desired_vector_.normalized() * speed;
110  }
