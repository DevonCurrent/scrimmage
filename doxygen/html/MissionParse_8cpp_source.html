65 void MissionParse::set_overrides(const std::string &overrides) {
66  std::string overrides_no_space = remove_whitespace(overrides);
67 
68  // Iterate over comma-separated overrides.
69  std::vector<std::string> overrides_tokens;
70  split(overrides_tokens, overrides_no_space, ", ");
71  for (auto &overrides_str: overrides_tokens) {
72  // Parse each key=value pair
73  std::vector<std::string> kv_tokens;

75 
76  // Only add to the map if the key:value was parsed correctly
77  if (kv_tokens.size() == 2) {

79  }

620  // Tokenize rate based on "/"
621  std::vector<double> values;
622  bool status = str2container(script_info["generate_rate"], "/",
623  values, 2);
624  double rate = -1;
625  if (status) {
626  rate = values[0] / values[1];
627  } else {
628  rate = std::stod(script_info["generate_rate"]);
629  }
630 
631  double start_time = 0;
632  if (script_info.count("generate_start_time") > 0) {
633  start_time = std::stod(script_info["generate_start_time"]);
634  }
635 
636  if (script_info.count("generate_time_variance") > 0) {
637  gen_info.time_variance = std::stod(script_info["generate_time_variance"]);
638  }
639 
640  int gen_count = std::stoi(script_info["generate_count"]);
641  if (rate > 0 && gen_count > 0) {
642  gen_info.start_time = start_time;
643  gen_info.gen_count = gen_count;
644  gen_info.rate = rate;
645 
646  } else if (rate > 0 && gen_count <= 0) {
647  cout << "WARNING: Not using entity generator."
648  << "generate_rate defined, but generate_count is "
649  << "less than or equal to zero" << endl;
650  }
651  }
652 
653  std::vector<double> start_times;
654  for (int i = 0; i < gen_info.gen_count; i++) {
655  start_times.push_back(gen_info.start_time);
656  }
657  next_gen_times_[ent_desc_id] = start_times;
658  gen_info_[ent_desc_id] = gen_info;
659 
660  // If the entity block has a "tag" attribute, save the mapping from the
661  // tag to the entity block ID
662  rapidxml::xml_attribute<> *tag_attr = script_node->first_attribute("tag");
663  if (tag_attr != 0) {
664  entity_tag_to_id_[tag_attr->value()] = ent_desc_id;
665  }
666 
667  entity_descs_[ent_desc_id++] = script_info;
668  }
669 
670  parse_terrain();
671 
672  // Parse the output_type
673  auto it_output_type = params_.find("output_type");
674  if (it_output_type != params_.end()) {
675  output_types_ = str2container<std::set<std::string>>(it_output_type->second,
676  ", ");
677  }
678  // If "all" is a possible output_type, include all output_types
679  if (output_types_.find("all") != output_types_.end()) {
680  output_types_ = possible_output_types_;
681  }
682 
683  return true;

686 bool MissionParse::create_log_dir() {
687  // Create the root_log_dir_ if it doesn't exist:
688  if (not fs::exists(fs::path(root_log_dir_)) &&
689  not fs::create_directories(fs::path(root_log_dir_))) {
690  cout << "Failed to create the root_log_dir: " << root_log_dir_ << endl;
691  return false;
692  }
693 
694  bool log_dir_created = false;
695  std::string log_dir_original = log_dir_;
696  int attempts = 0;
697  while (!log_dir_created && attempts < 1e4) {
698  log_dir_ = log_dir_original;
699  // If the log directory already exists, append a number to it:
700  if (fs::exists(fs::path(log_dir_))) {
701  int ct = 1;
702  std::string log_dir_tmp;
703  do {
704  log_dir_tmp = log_dir_ + "_" + std::to_string(ct++);
705  } while (fs::exists(fs::path(log_dir_tmp)));
706  log_dir_ = log_dir_tmp;
707  }
708 
709  if (fs::create_directories(fs::path(log_dir_))) {
710  log_dir_created = true;
711  }
712  attempts++;
713  }
714 
715  if (!log_dir_created) {
716  cout << "Unable to create log directory: " << log_dir_ << endl;
717  return false;
718  }
719 
720  // Copy the input scenario xml file to the output directory
721  if (fs::exists(mission_filename_)) {
722  fs::copy_file(fs::path(mission_filename_), fs::path(log_dir_+"/mission.orig.xml"));
723  }
724 
725  // Write the parsed content to the output directory
726  std::ofstream content_out(log_dir_+"/mission.xml");
727  content_out << mission_file_content_;
728  content_out.close();
729 
730  // Create the latest log directory by default. Don't create the latest
731  // directory if the tag is defined in the mission file and it is set to
732  // false.
733  bool create_latest_dir = not(params_.count("create_latest_dir") > 0 &&
734  str2bool(params_["create_latest_dir"]) == false);
735 
736  if (create_latest_dir) {
737  boost::system::error_code ec;
738  auto print_error = [&]() {
739  cout << "Error code value: " << ec.value() << endl;
740  cout << "Error code name: " << ec.category().name() << endl;
741  cout << "Error message: " << ec.message() << endl;
742  };
743  auto fs_err = [&](){return ec != boost::system::errc::success;};
744 
745  // Create a "latest" symlink to the directory
746  // First, remove the latest symlink if it exists
747  fs::path latest_sym(root_log_dir_ + std::string("/latest"));
748  if (fs::is_symlink(latest_sym)) {
749  fs::remove(latest_sym, ec);
750  if (fs_err()) {
751  cout << "WARNING: could not remove symlink to latest directory" << endl;
752  print_error();
753  }
754  }
755 
756  // Create the symlink
757  fs::create_directory_symlink(fs::path(log_dir_), latest_sym, ec);
758  if (fs_err()) {
759  cout << "WARNING: Unable to create latest log file symlink" << endl;
760  cout << "Couldn't create symlink log directory: "
761  << latest_sym.string() << " -> "
762  << fs::path(log_dir_).string() << endl;
763  print_error();
764  }
765  }
766  return true;
767 }

791 bool MissionParse::parse_terrain() {
792  ConfigParse terrain_parse;
793  utm_terrain_ = std::make_shared<scrimmage_proto::UTMTerrain>();
794 
795  // Check for grid settings
796  if (params_.count("grid_spacing") > 0) {
797  utm_terrain_->set_grid_spacing(std::stod(params_["grid_spacing"]));
798  } else {
799  utm_terrain_->set_grid_spacing(100);
800  }
801 
802  if (params_.count("grid_size") > 0) {
803  utm_terrain_->set_grid_size(std::stod(params_["grid_size"]));
804  } else {
805  utm_terrain_->set_grid_size(10000);
806  }
807 
808  // Check for origin settings
809  if (params_.count("origin_length") > 0) {
810  utm_terrain_->set_origin_length(std::stod(params_["origin_length"]));
811  } else {
812  utm_terrain_->set_origin_length(1);
813  }
814 
815  if (params_.count("show_origin") > 0) {
816  utm_terrain_->set_show_origin(str2bool(params_["show_origin"]));
817  } else {
818  utm_terrain_->set_show_origin(false);
819  }
820 
821  set(utm_terrain_->mutable_background(), background_color_);
822 
823  utm_terrain_->set_enable_grid(true);
824  utm_terrain_->set_enable_terrain(false);
825  utm_terrain_->set_enable_extrusion(false);
826 
827  if (params_.count("terrain") > 0 &&
828  find_terrain_files(params_["terrain"], terrain_parse, utm_terrain_)) {
829 
830  double x_easting;
831  double y_northing;
832  int zone;
833  bool northp;
834  double gamma, k_scale;
835  GeographicLib::UTMUPS::Forward(latitude_origin(),
836  longitude_origin(),
837  zone, northp, x_easting, y_northing,
838  gamma, k_scale);
839 
840  // Make sure the projected zone and hemisphere match the input xml
841  if (((northp && boost::to_upper_copy(terrain_parse.params()["hemisphere"]) == "NORTH") ||
842  (!northp && boost::to_upper_copy(terrain_parse.params()["hemisphere"]) == "SOUTH")) &&
843  zone == get("zone", terrain_parse.params(), -2)) {
844 
845  double origin_offset_x = 0;
846  double origin_offset_y = 0;
847  double origin_offset_z = 0;
848 
849  if (terrain_parse.params().count("origin_offset_x") == 1
850  && terrain_parse.params().count("origin_offset_y") == 1
851  && terrain_parse.params().count("origin_offset_z") == 1) {
852 
853  origin_offset_x = stod(terrain_parse.params()["origin_offset_x"]);
854  origin_offset_y = stod(terrain_parse.params()["origin_offset_y"]);
855  origin_offset_z = stod(terrain_parse.params()["origin_offset_z"]);
856  }
857 
858  utm_terrain_->set_x_translate(x_easting - origin_offset_x);
859  utm_terrain_->set_y_translate(y_northing - origin_offset_y);
860  utm_terrain_->set_z_translate(altitude_origin() - origin_offset_z);
861  utm_terrain_->set_z_translate(altitude_origin());
862  utm_terrain_->set_system(terrain_parse.params()["system"]);
863  utm_terrain_->set_zone(zone);
864  utm_terrain_->set_hemisphere(terrain_parse.params()["hemisphere"]);
865  utm_terrain_->set_enable_grid(get<bool>("enable_grid", terrain_parse.params(), "false"));
866  utm_terrain_->set_enable_terrain(true);
867  return true;
868 
869  } else {
870  cout << "============================================" << endl;
871  cout << "Invalid XML Terrain settings: " << endl;
872  cout << "Hemisphere: " << terrain_parse.params()["hemisphere"] << endl;
873  cout << "Zone: " << get("zone", terrain_parse.params(), -2) << endl;
874  cout << "--------------------------------------------" << endl;
875  cout << "Geographic lib output: " << endl;
876  cout << "x_easting: " << x_easting << endl;
877  cout << "y_northing: " << y_northing << endl;
878  cout << "Zone: " << zone << endl;
879  cout << "Northern hemisphere?: " << northp << endl;
880  }
881  }
882  return false;
883 }
