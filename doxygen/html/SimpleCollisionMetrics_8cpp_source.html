121 void SimpleCollisionMetrics::calc_team_scores() {
122  double end_time = -std::numeric_limits<double>::infinity();
123  double beg_time = std::numeric_limits<double>::infinity();
124  for (auto &kv : scores_) {
125  double temp_beg_time = kv.second.flight_time_start();
126  double temp_end_time = kv.second.flight_time_end();
127  if (temp_beg_time < beg_time) {
128  beg_time = temp_beg_time;
129  }
130  if (temp_end_time > end_time) {
131  end_time = temp_end_time;
132  }
133  }
134 
135  // If the end_time is <= beginning time, all entities survived, need
136  // to use the ending simulation time.
137  if (end_time <= beg_time) {
138  end_time = time_->t();
139  }
140 
141  double max_flight_time = end_time - beg_time;
142 
143  // Create the score, if necessary
144  for (auto &team_id : teams_) {
145  if (team_coll_scores_.count(team_id) == 0) {
146  SimpleCollisionScore score;
147  score.set_weights(params_);
148  // Set the max flight time for each score:
149  score.set_max_flight_time(max_flight_time);
150  team_coll_scores_[team_id] = score;
151  }
152  }
153 
154  for (auto &kv : scores_) {
155  SimpleCollisionScore &score = kv.second;
156 
157  // If the end time is less than start time, the flight time was never
158  // set. Use the ending simulation time.
159  if (score.flight_time_end() <= score.flight_time_start()) {
160  score.set_flight_time_end(end_time);
161  }
162 
163  int team_id = (*id_to_team_map_)[kv.first];
164 
165  team_coll_scores_[team_id].add_non_team_collisions(
166  score.non_team_collisions());
167  team_coll_scores_[team_id].add_team_collisions(
168  score.team_collisions());
169  team_coll_scores_[team_id].add_ground_collisions(
170  score.ground_collisions());
171  team_coll_scores_[team_id].add_flight_time(score.flight_time());
172  team_coll_scores_[team_id].increment_entity_count();
173  }
174 
175  for (auto &kv : team_coll_scores_) {
176  int team_id = kv.first;
177  SimpleCollisionScore &score = kv.second;
178  team_metrics_[team_id]["entity_count"] = score.entity_count();
179  team_metrics_[team_id]["flight_time"] = score.flight_time();
180  team_metrics_[team_id]["flight_time_norm"] = score.flight_time_norm();
181  team_metrics_[team_id]["non_team_coll"] = score.non_team_collisions();
182  team_metrics_[team_id]["team_coll"] = score.team_collisions();
183  team_metrics_[team_id]["ground_coll"] = score.ground_collisions();
184 
185  team_scores_[team_id] = score.score();
186  }
187 
188  // list the headers we want put in the csv file
189  headers_.push_back("entity_count");
190  headers_.push_back("flight_time");
191  headers_.push_back("flight_time_norm");
192  headers_.push_back("non_team_coll");
193  headers_.push_back("team_coll");
194  headers_.push_back("ground_coll");
195 }
