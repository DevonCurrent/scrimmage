86 void AirSimSensor::parse_camera_configs(std::map<std::string, std::string> &params) {
87  // Parse the camera config string.
88  // The string is a list of camera configs from AirSimSensor.xml of the form:
89  // [VehicleName=robot1 CameraName=front_center ImageTypeName=Scene Width=256 Height=144]
90  std::string camera_config = sc::get<std::string>("camera_config", params, "");
91  std::vector<std::string> tokens_1;
92  boost::split(tokens_1, camera_config, boost::is_any_of("[]"));
93  for (std::string &t_1 : tokens_1) {
94  if (t_1.length() > 0) {
95  std::vector<std::string> tokens_2;
96  boost::split(tokens_2, t_1, boost::is_any_of(" ,"));
97 
98  if ((tokens_2.size() == 6) && (tokens_2[0] == vehicle_name_)) {
99 
100  try {
101  CameraConfig c;
102  c.vehicle_name = tokens_2[0];
103  c.cam_name = tokens_2[1];
104 
105  // Get Image Type Name and Number
106  // Scene=0, DepthPlanner=1, DepthPerspective=2, DepthVis=3, DisparityNormalized=4,
107  // Segmentation=5, SurfaceNormals=6, Infrared=7
108  if (tokens_2[2] == "Scene") {
109  c.img_type_number = 0;
110  c.img_type = ma::ImageCaptureBase::ImageType::Scene;
111  c.img_type_name = "Scene";
112  } else if (tokens_2[2] == "DepthPlanner") {
113  c.img_type_number = 1;
114  c.img_type = ma::ImageCaptureBase::ImageType::DepthPlanner;
115  c.img_type_name = "DepthPlanner";
116  } else if (tokens_2[2] == "DepthPerspective") {
117  c.img_type_number = 2;
118  c.img_type = ma::ImageCaptureBase::ImageType::DepthPerspective;
119  c.img_type_name = "DepthPerspective";
120  } else if (tokens_2[2] == "DepthVis") {
121  c.img_type_number = 3;
122  c.img_type = ma::ImageCaptureBase::ImageType::DepthVis;
123  c.img_type_name = "DepthVis";
124  } else if (tokens_2[2] == "DisparityNormalized") {
125  c.img_type_number = 4;
126  c.img_type = ma::ImageCaptureBase::ImageType::DisparityNormalized;
127  c.img_type_name = "DisparityNormalized";
128  } else if (tokens_2[2] == "Segmentation") {
129  c.img_type_number = 5;
130  c.img_type = ma::ImageCaptureBase::ImageType::Segmentation;
131  c.img_type_name = "Segmentation";
132  } else if (tokens_2[2] == "SurfaceNormals") {
133  c.img_type_number = 6;
134  c.img_type = ma::ImageCaptureBase::ImageType::SurfaceNormals;
135  c.img_type_name = "SurfaceNormals";
136  } else if (tokens_2[2] == "Infrared") {
137  c.img_type_number = 7;
138  c.img_type = ma::ImageCaptureBase::ImageType::Infrared;
139  c.img_type_name = "Infrared";
140  } else {
141  cout << "Error: Unknown image type: " << tokens_2[2] << endl;
142  c.img_type = ma::ImageCaptureBase::ImageType::Scene;
143  c.img_type_name = "Scene";
144  }
145 
146  // Get Image pixel width and height (integers)
147  c.width = std::stoi(tokens_2[3]);
148  c.height = std::stoi(tokens_2[4]);
149  c.fov = std::stoi(tokens_2[5]);
150 
151  cout << "[AirSimSensor] Adding camera to Vehicle '" << vehicle_name_ << "' " << c << endl;
152  cam_configs_.push_back(c);
153  } catch (boost::bad_lexical_cast) {
154  // Parsing whitespace and possibily malformed XML.
155  cout << "Parse Error: Check camera_config's in AirSimSensor.xml" << endl;
156  } // end try, catch
157  }
158  } // end check token_1 size

313  // Set up stream of camera images to be published to scrimmage messages.
314  t_start = std::chrono::high_resolution_clock::now();
315  // Make the Requests Vector for getting images
316  vector<ImageRequest> requests;
317  for (CameraConfig c : cam_configs_) {
318  // Depth Images (Depth Perspective and Depth Planner) come in as 1 channel float arrays
319  ImageRequest request;
320  if (c.img_type_name == "DepthPerspective" || c.img_type_name == "DepthPlanner") {
321  request = {ImageRequest(c.cam_name, c.img_type, true, false)};
322  } else {
323  // All Other Image types come in as RGB = 3 channel uint8 matrix
324  request = {ImageRequest(c.cam_name, c.img_type, false, false)};
325  }
326  requests.push_back(request);
327  }
328 
329  // Get Images
330  const std::vector<ImageResponse>& response_vector = img_client->simGetImages(requests, vehicle_name_);
331  // request vehicle pose to ensure it is up to date with camera pose
332  ma::Pose vehicle_pose_camera = img_client->simGetVehiclePose(vehicle_name_);
333  // AirSim vehicle pose is in NED, but scrimmage is in ENU so convert
334  // Convert position to ENU: Switch X and Y and negate Z
335 
336  // If response vector contains new data, create img_msg, transfer data into message and publish
337  // cout << response_vector.size() << endl;
338  if (response_vector.size() > 0) {
339  auto im_msg = std::make_shared<sc::Message<std::vector<AirSimImageType>>>();
340  // sc::MessagePtr<std::vector<AirSimImageType>> im_msg;
341 
342  for (ImageResponse response : response_vector) {
343  CameraConfig response_cam_config;
344  for (CameraConfig c : cam_configs_) {
345  if (c.img_type == response.image_type && c.cam_name == response.camera_name) {
346  response_cam_config = c;
347  break;
348  }
349  }
350  // Create AirSimImageType variable
351  AirSimImageType a;
352  a.camera_config = response_cam_config;
353  a.vehicle_name = vehicle_name_;
354 
355  // AirSim gives pose of vehicle in relation to the world frame in NED
356  Eigen::Translation3f translation_trans_vehicle(vehicle_pose_camera.position);
357  Eigen::Quaternionf rotation_quat_vehicle(vehicle_pose_camera.orientation);
358  Eigen::Isometry3f tf_world_vehicle_NED(translation_trans_vehicle * rotation_quat_vehicle);
359  a.vehicle_pose_world_NED = tf_world_vehicle_NED;
360  // AirSim gives pose of camera in relation to the world frame in NED
361  Eigen::Translation3f translation_trans_camera(response.camera_position);
362  Eigen::Quaternionf rotation_quat_camera(response.camera_orientation);
363  Eigen::Isometry3f tf_world_camera_NED(translation_trans_camera * rotation_quat_camera);
364  a.camera_pose_world_NED = tf_world_camera_NED;
365 
366  // Depth Images (Depth Perspective and Depth Planner) come in as 1 channel float arrays
367  a.camera_config.pixels_as_float = response.pixels_as_float;
368  if (a.camera_config.pixels_as_float) {
369  // get uncompressed 32FC1 array bytes
370  a.img = cv::Mat(a.camera_config.height, a.camera_config.width, CV_32FC1);

372 
373  } else {

375  // However old linux asset environments give 4 channels
376 
377  // If image has 4 channels

388  } // end ImageResponse for loop

398  } // end response > 0

404  t_end = std::chrono::high_resolution_clock::now();
405  double t_elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(t_end - t_start).count();

597  cout << "X" << std::flush;
598  std::this_thread::sleep_for(std::chrono::seconds(1));
599  // If we haven't been able to connect to AirSim, warn the user and return.
600  if (i == 10) {

606  std::make_shared<ma::MultirotorRpcLibClient>(airsim_ip_,
607  airsim_port_,
608  airsim_timeout_s_);
609  } else {
610  client_connected_ = true;
611  cout << "[AirSimSensor] Sim Client for " << vehicle_name_ << " connected to AirSim: ip " << airsim_ip_ << ", port " << airsim_port_ << endl;
612  break;

614  }
615 
616  sim_client_->enableApiControl(true, vehicle_name_);
617  }
618 
622 
623  // Setup state information for AirSim
624  sc::StatePtr &state = parent_->state_truth();

634  state->quat().roll(),
635  airsim_yaw_rad);
636

639

641  if (get_image_data_) {
642  // sc::MessagePtr<std::vector<AirSimImageType>> im_msg_step;

644

647  new_image_ = false;
648  new_image_mutex_.unlock();

653  // sc::MessagePtr<std::vector<AirSimImageType>> im_msg_step = img_msg_;
654  img_msg_mutex_.unlock();
655

661  img_pub_->publish(im_msg_step);
662  }
663  }

668

671  new_lidar_ = false;
672  new_lidar_mutex_.unlock();
673

675  std::shared_ptr<sc::Message<std::vector<AirSimLidarType>>> lidar_msg_step = lidar_msg_;
676  // lidar_msg_step = lidar_msg_; // NOLINT
677  lidar_msg_mutex_.unlock();
678 
679  // If lidar is new, publish
680  if (new_lidar) {
681  lidar_pub_->publish(lidar_msg_step);
682  }
683  }
684 
685  if (get_imu_data_) {
686  // auto imu_msg_step = std::make_shared<sc::Message<std::vector<AirSimImuType>>>(); // NOLINT
687 
688  new_imu_mutex_.lock();
689  bool new_imu = new_imu_;
690  new_imu_ = false;
691  new_imu_mutex_.unlock();
692 
693  imu_msg_mutex_.lock();
694  std::shared_ptr<sc::Message<std::vector<AirSimImuType>>> imu_msg_step = imu_msg_;

709 bool AirSimSensor::save_data(MessagePtr<std::vector<AirSimImageType>>& im_msg, sc::StatePtr& state, int frame_num) {
710  // Get timestamp
711  double time_now = time_->t();
712 
713  // TODO: Saving 3 images + a csv makes the simulation run slower, maybe need to thread
714  for (AirSimImageType d : im_msg->data) {
