207 void ContactBlobCamera::draw_frustum(const std::vector<scrimmage_proto::ShapePtr>& frustum_shapes, double x_rot, double y_rot, double z_rot) {
208  if (frustum_shapes.size() != 8) {
209  std::cerr << "ContactBlobCamera::draw_frustum: ERROR: input shape vector must be size 8" << std::endl;
210  return;
211  }
212  double sensor_footprint_height = max_detect_range_ * tan(el_thresh_ / 2) * 2;
213  double sensor_footprint_width = max_detect_range_ * tan(az_thresh_ / 2) * 2;
214  Eigen::Vector3d sensor_UL(max_detect_range_, sensor_footprint_width / 2, -sensor_footprint_height / 2);
215  Eigen::Vector3d sensor_UR(max_detect_range_, sensor_footprint_width / 2, sensor_footprint_height / 2);
216  Eigen::Vector3d sensor_LR(max_detect_range_, -sensor_footprint_width / 2, sensor_footprint_height / 2);
217  Eigen::Vector3d sensor_LL(max_detect_range_, -sensor_footprint_width / 2, -sensor_footprint_height / 2);
218  std::vector<Eigen::Vector3d> sensor_scene_bb = {sensor_UL, sensor_UR, sensor_LR, sensor_LL, sensor_UL};
219 
220  // Rotate into sensor frame
221  auto x_rotation = Eigen::AngleAxisd(x_rot, Eigen::Vector3d::UnitX());
222  auto y_rotation = Eigen::AngleAxisd(y_rot, Eigen::Vector3d::UnitY());
223  auto z_rotation = Eigen::AngleAxisd(z_rot, Eigen::Vector3d::UnitZ());
224  auto A = Eigen::Translation3d(parent_->state_truth()->pos()) * z_rotation * y_rotation * x_rotation;
225  for (size_t i = 0; i < sensor_scene_bb.size(); i++) {
226  sensor_scene_bb[i] = A * sensor_scene_bb[i];
227  }
228 
229  // Draw scene box
230  auto scene_bb_line = std::make_shared<sp::Shape>();
231  scene_bb_line->set_opacity(1.0);
232  scene_bb_line->set_persistent(false);
233  scene_bb_line->set_persist_duration(0.0);
234  sc::set(scene_bb_line->mutable_color(), 0, 255, 0);
235 
236  sc::path_to_lines(sensor_scene_bb, scene_bb_line,
237  std::static_pointer_cast<EntityPlugin>(shared_from_this()));
238  // draw 8 lines for the 4 points
239  const Eigen::Vector3d fov_edge_start = parent_->state_truth()->pos();
240  for (int idx_bb = 0; idx_bb != 4; ++idx_bb) {
241  // box edges are idx 0-3 in frustum, fov edges are 4-7
242  const size_t idx_fov_edge = idx_bb + 4;
243 
244  // set box line
245  Eigen::Vector3d box_edge_start;
246  const Eigen::Vector3d box_edge_end = sensor_scene_bb.at(idx_bb);
247  if (idx_bb == 0) {
248  box_edge_start = sensor_scene_bb.back();
249  } else {
250  box_edge_start = sensor_scene_bb.at(idx_bb-1);
251  }
252  scrimmage::set(frustum_shapes.at(idx_bb)->mutable_line()->mutable_start(), box_edge_start);
253  scrimmage::set(frustum_shapes.at(idx_bb)->mutable_line()->mutable_end(), box_edge_end);
254 
255  // set fov line
256  scrimmage::set(frustum_shapes.at(idx_fov_edge)->mutable_line()->mutable_start(), fov_edge_start);
257  scrimmage::set(frustum_shapes.at(idx_fov_edge)->mutable_line()->mutable_end(), box_edge_end);
258  }
259  // draw all lines
260  for (auto&& shape : frustum_shapes) {
261  draw_shape(shape);
262  }

265 void ContactBlobCamera::contacts_to_bounding_boxes(
266  const scrimmage::State &sensor_frame,
267  scrimmage::ContactMap &contacts,
268  std::shared_ptr<sc::Message<ContactBlobCameraType>> &msg) {
269 
270  for (auto &kv : contacts) {
271  // Filter out (skip) own contact
272  if (kv.second.id().id() == parent_->id().id()) continue;
273 
274  // Filter out contacts out of range, should use RTree, but rtree still
275  // requires querying of ID from contact lists. (TODO)
276  if ((kv.second.state()->pos() - sensor_frame.pos()).norm() >
277  max_detect_range_) {
278  continue;
279  }
280 
281  // Don't "detect" current contact relative to false negative probability
282  double r = parent_->random()->rng_uniform(0.0, 1.0);
283  if (r < fn_prob_) continue;
284 
285  // Transform contact into "camera" coordinate system
286  Eigen::Vector3d rel_pos = sensor_frame.rel_pos_local_frame(kv.second.state()->pos());
287 
288  if (!in_field_of_view(rel_pos)) continue;
289 
290  // Convert 3D relative position to 2d image plane position
291  Eigen::Vector2d raster_center = project_rel_3d_to_2d(rel_pos);
292 
293  double object_radius = kv.second.radius();
294 
295  // Get angle between (vector between camera and object center) and
296  // (vector between object center and line tangent to object circle)
297  double beta = acos(object_radius / rel_pos.norm());
298 
299  // Get vector pointing from contact to camera
300  Eigen::Vector2d v_r(-rel_pos(0), -rel_pos(1));
301 
302  // Normalize to unit vector and Make length equal to object radius
303  v_r = v_r.normalized() * object_radius;
304 
305  // Rotate one direction and add circle's center point
306  Eigen::Vector2d v_r_1 = v_r, v_r_2 = v_r;
307  Eigen::Rotation2D<double> rot1(beta);
308  Eigen::Rotation2D<double> rot2(-beta);
309  v_r_1 = rot1.toRotationMatrix() * v_r_1;
310  v_r_2 = rot2.toRotationMatrix() * v_r_2;
311 
312  // Get 3D relative position of rotated vectors
313  Eigen::Vector3d p1(rel_pos(0) + v_r_1(0),
314  rel_pos(1) + v_r_1(1),
315  rel_pos(2));
316 
317  Eigen::Vector3d p2(rel_pos(0) + v_r_2(0),
318  rel_pos(1) + v_r_2(1),
319  rel_pos(2));
320 
321  // Get 2D position of object boundaries
322  Eigen::Vector2d r1 = project_rel_3d_to_2d(p1);
323  Eigen::Vector2d r2 = project_rel_3d_to_2d(p2);
324 
325  // Calculate image radius using distance between object boundaries
326  double object_img_radius = (r1 - r2).norm() / 2.0;
327 
328  double object_img_half_width = std::max(1.0, object_img_radius + (*size_noise_[0])(*gener_));
329  double object_img_half_height = std::max(1.0, object_img_radius + (*size_noise_[1])(*gener_));
330 
331  // Add noise to position in 2D image plane
332  raster_center(0) += (*pos_noise_[0])(*gener_);
333  raster_center(1) += (*pos_noise_[1])(*gener_);
334 
335  // Add bounding box to frame around object
336  cv::Rect rect(std::floor(raster_center(0))-std::floor(object_img_half_width),
337  std::floor(raster_center(1))-std::floor(object_img_half_height),
338  std::floor(object_img_half_width*2) + 1,
339  std::floor(object_img_half_height*2) + 1);
340 
341  if (object_img_radius > 0) {
342  draw_object_with_bounding_box(msg->data.frame, kv.second.id().id(),
343  rect, raster_center, object_img_radius);
344  }
345 
346  // Collect all bounding boxes for current detected object
347  msg->data.bounding_boxes[kv.second.id().id()].push_back(rect);
348  }

469 Eigen::Vector2d ContactBlobCamera::project_rel_3d_to_2d(Eigen::Vector3d rel_pos) {
470  // 3D to 2D transforms are described here:
471  // http://www.scratchapixel.com/lessons/3d-basic-rendering/computing-pixel-coordinates-of-3d-point/mathematics-computing-2d-coordinates-of-3d-points
472 
473  // Convert relative position of contact into computer vision coordinates
474  // for relative coordinates:
475  Eigen::Vector3d pos(rel_pos(1), rel_pos(2), rel_pos(0));
476 
477  // Transform to screen / canvas space
478  Eigen::Vector2d s(focal_length_ * -pos(0) / pos(2),
479  focal_length_ * pos(1) / pos(2));
480 
481  // Transform to Normalized Device Coordinate (NDC) space
482  Eigen::Vector2d ndc((s(0) + canvas_width_/2.0) / canvas_width_,
483  (s(1) + canvas_height_/2.0) / canvas_height_);
484 
485  // Tranform to raster space
486  Eigen::Vector2d r(ndc(0) * img_width_,
487  (1-ndc(1)) * img_height_);
488  return r;
489 }

491 bool ContactBlobCamera::in_field_of_view(Eigen::Vector3d rel_pos) {
492  double az = atan2(rel_pos(1), rel_pos(0));
493  double norm_xy = sqrt(pow(rel_pos(0), 2) + pow(rel_pos(1), 2));
494  double el = atan2(rel_pos(2), norm_xy);
495  if (std::abs(az) > (az_thresh_ / 2) ||
496  std::abs(el) > (el_thresh_ / 2)) {
497  return false;
498  }
499  return true;
500 }
