161 void RigidBody6DOF::model(const vector_t &x , vector_t &dxdt , double t) {
162  double thrust = (*ctrl_u_)(THRUST);
163  double elevator = (*ctrl_u_)(ELEVATOR);
164  double aileron = (*ctrl_u_)(AILERON);
165  double rudder = (*ctrl_u_)(RUDDER);
166 
167  double F_thrust = thrust;
168 
169  double F_x = F_thrust;
170 
171  double m = 1.0;
172  double g = 0; // -9.8; no gravity
173  double theta = 0;
174 
175  double Ixx = 1;
176  double Iyy = 1;
177  double Izz = 1;
178  double Ixz = 0;
179 
180  // TODO: Should these be cached from previous run or should the current dxdt
181  // versions be used?
182  double U_dot = dxdt[U];
183  double V_dot = dxdt[V];
184  double W_dot = dxdt[W];
185  double P_dot = dxdt[P];
186  double Q_dot = dxdt[Q];
187  double R_dot = dxdt[R];
188 
189  dxdt[U] = x[V]*x[R] - x[W]*x[Q] - g*sin(theta) + F_x / m;
190  dxdt[V] = 0; // x[W]*x[P] - x[U]*x[R] + g*sin(phi)*cos(theta) + F_y / m;
191  dxdt[W] = 0; // x[U]*x[Q] - x[V]*x[P] + g*cos(phi)*cos(theta) + F_z / m;
192 
193  double L = aileron + Ixx*P_dot - Ixz*R_dot - Ixz*x[P]*x[Q] + (Izz - Iyy)*x[R]*x[Q];
194  double M = elevator + Iyy*Q_dot + (Ixx-Izz)*x[P]*x[R] + (Ixz*(pow(x[P], 2) - pow(x[R], 2)));
195  double N = rudder + Izz*R_dot - Ixz*P_dot + (Iyy-Ixx)*x[P]*x[Q] + Ixz*x[Q]*x[R];
196 
197  double L_tic = L + Ixz*x[P]*x[Q] - (Izz-Iyy)*x[R]*x[Q];
198  double N_tic = N - (Iyy-Ixx)*x[P]*x[Q] - Ixz*x[R]*x[Q];
199 
200  dxdt[P] = (L_tic*Izz-N_tic*Ixz) / (Ixx*Izz - pow(Ixz, 2));
201  dxdt[Q] = (M - (Ixx - Izz)*x[P]*x[R]) - Ixz*(pow(x[P], 2)-pow(x[R], 2))/Iyy;
202  dxdt[R] = (N_tic*Ixx+L_tic*Ixz) / (Ixx*Izz - pow(Ixz, 2));
203 
204  double lambda = 1 - (pow(x[q0], 2) + pow(x[q1], 2) + pow(x[q2], 2) + pow(x[q3], 2));
205  dxdt[q0] = -0.5 * (x[q1]*x[P] + x[q2]*x[Q] + x[q3]*x[R]) + lambda * x[q0];
206  dxdt[q1] = +0.5 * (x[q0]*x[P] + x[q2]*x[R] - x[q3]*x[Q]) + lambda * x[q1];
207  dxdt[q2] = +0.5 * (x[q0]*x[Q] + x[q3]*x[P] - x[q1]*x[R]) + lambda * x[q2];
208  dxdt[q3] = +0.5 * (x[q0]*x[R] + x[q1]*x[Q] - x[q2]*x[P]) + lambda * x[q3];
209 
210  // Local position / velocity to global
211  // Normalize quaternion
212  sc::Quaternion quat(x[q0], x[q1], x[q2], x[q3]);
213  quat.w() = x[q0];
214  quat.x() = x[q1];
215  quat.y() = x[q2];
216  quat.z() = x[q3];
217 
218  quat = quat_world_ * quat;
219  quat.normalize();
220 
221  // Convert local positions and velocities into global coordinates
222  Eigen::Matrix3d rot = quat.toRotationMatrix();
223 
224  Eigen::Vector3d vel_local(x[U], x[V], x[W]);
225  Eigen::Vector3d vel_world = rot * vel_local;
226  dxdt[Xw] = vel_world(0);
227  dxdt[Yw] = vel_world(1);
228  dxdt[Zw] = vel_world(2);
229 
230  Eigen::Vector3d acc_local(U_dot, V_dot, W_dot);
231  Eigen::Vector3d acc_world = rot * acc_local;
232  dxdt[Uw] = acc_world(0);
233  dxdt[Vw] = acc_world(1);
234  dxdt[Ww] = acc_world(2);
235 }
