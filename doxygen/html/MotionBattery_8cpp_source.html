117 bool MotionBattery::step(double t, double dt) {
118  if (publish_charge_) {

120  pub_charge_percentage_->publish(msg);
121  }

123  // Loop over all variables in the io_map_, apply charge calculation, and
124  // limit output if required
125  for (auto &kv : io_map_) {
126  double value = vars_.input(kv.second->input_idx);
127  double depletion = kv.second->calc_depletion(value, time_->dt());
128  if (!battery_.deplete(depletion)) {
129  // If deplete() returns false, the battery is depleted.
130  parent()->collision();
