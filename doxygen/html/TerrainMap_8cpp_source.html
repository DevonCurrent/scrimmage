104 bool TerrainMap::generate_linear() {
105  // Force the altitude center to be at the midpoint between z_max and z_min
106  center_(2) = (z_max_ + z_min_) / 2.0;
107 
108  // Initialize the map, such that row 0 is at z_min and the last row is at
109  // z_max, with a linear interpolation across the rows. Also, add noise to
110  // each height value.
111  double z_step = (z_max_ - z_min_) / num_y_rows_;
112 
113  for (unsigned int row = 0; row < num_y_rows_; ++row) {
114  double height = z_step * row;
115  for (unsigned int col = 0; col < num_x_cols_; ++col) {
116  grid_[row][col].height = height + (*rng_)(*gener_);
117  grid_[row][col].is_set = true;
118  }
119  }
120  center_height_adjust();
121  clamp_height();
122  return true;
123 }

196 double TerrainMap::get_neighbor_avg(const int &row, const int &col) {
197  unsigned int neighbors = 0;
198  double height_sum = 0;
199 
200  if (row > 0) {
201  height_sum += grid_[row-1][col].height;
202  ++neighbors;
203  }
204  if (col > 0) {
205  height_sum += grid_[row][col-1].height;
206  ++neighbors;
207  }
208  if (row > 0 && col > 0) {
209  height_sum += grid_[row-1][col-1].height;
210  ++neighbors;
211  }
212 
213  return (neighbors == 0) ? 0 : height_sum / neighbors;
214 }

254 boost::optional<double> TerrainMap::height_at(const double &x, const double &y) {
