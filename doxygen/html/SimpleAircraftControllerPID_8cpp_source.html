80 bool SimpleAircraftControllerPID::step(double t, double dt) {
81  heading_pid_.set_setpoint(vars_.input(input_roll_or_heading_idx_));
82  double u_roll_rate = use_roll_ ?
83  -heading_pid_.step(dt, state_->quat().roll()) :
84  heading_pid_.step(dt, state_->quat().yaw());
85 
86  alt_pid_.set_setpoint(vars_.input(input_altitude_or_glide_slope_idx_));
87  double u_pitch_rate = use_glide_slope_ ?
88  alt_pid_.step(dt, atan(state_->vel()(2) / sqrt(state_->vel()(0) * state_->vel()(0) + state_->vel()(1) * state_->vel()(1)))) :
89  -alt_pid_.step(dt, state_->pos()(2));
90 
91  vel_pid_.set_setpoint(vars_.input(input_velocity_idx_));
92  double u_throttle = vel_pid_.step(dt, state_->vel().norm());
