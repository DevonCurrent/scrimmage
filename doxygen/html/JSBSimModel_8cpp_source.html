216 bool JSBSimModel::step(double time, double dt) {
217  double desired_velocity = vars_.input(speed_idx_);
218  double roll_cmd = vars_.input(roll_idx_);
219 
220  // + : roll right, - : roll left
221  bank_setpoint_node_->setDoubleValue(roll_cmd);
222  if (use_pitch_) {
223  double elevator_cmd = vars_.input(alt_or_pitch_idx_);
224 
225  // Negate altitude PID from the elevator control
226  elevator_cmd -= exec_->GetPropertyValue("ap/elevator_cmd");
227  fcs_elevator_cmd_node_->setDoubleValue(elevator_cmd);
228 
229  // Try to remove altitude control by putting setpoint to current altitude
230  desired_altitude_node_->setDoubleValue(state_->pos()(2) * meters2feet);
231  } else {
232  double desired_alt = vars_.input(alt_or_pitch_idx_);
233  // Set desired altitude (we just need the desired altitude, use the current
234  // x,y as placeholders).
235  double lat_curr, lon_curr, alt_result;
236  parent_->projection()->Reverse(state_->pos()(0), state_->pos()(1), desired_alt,
237  lat_curr, lon_curr, alt_result);
238 
239  desired_altitude_node_->setDoubleValue(alt_result * meters2feet);
240  }
241  // set desired velocity
242  desired_velocity_node_->setDoubleValue(desired_velocity * mps2knts);
243 
245  exec_->Setdt(dt);
246  exec_->Run();
247  // Save state
248 
249  if (fg_out_enable_) {
250  output_fg_->Print();
251  }
252 
253  parent_->projection()->Forward(latitude_node_->getDoubleValue(),
254  longitude_node_->getDoubleValue(),
255  altitude_node_->getDoubleValue() * feet2meters,
256  state_->pos()(0), state_->pos()(1), state_->pos()(2));
257 
258  angles_from_jsbsim_.set_angle(ang::rad2deg(yaw_node_->getDoubleValue()));
259 
260  state_->quat().set(roll_node_->getDoubleValue(),
261  -pitch_node_->getDoubleValue(),
262  ang::deg2rad(angles_from_jsbsim_.angle()));
263 
264  state_->vel() << vel_east_node_->getDoubleValue() * feet2meters,
265  vel_north_node_->getDoubleValue() * feet2meters,
266  -vel_down_node_->getDoubleValue() * feet2meters;
267 
268  // save what was used as the input
269  return true;
