83 bool BoundaryDefense::step_autonomy(double t, double dt) {
84  // Get the active boundary to defend
85  auto it = boundaries_.find(boundary_id_);
86  if (it == boundaries_.end()) {
87  return true;
88  }
89 
90  // Find the closest entity within the boundary that is not part of the
91  // boundary's team.
92  double min_dist = std::numeric_limits<double>::infinity();
93  sc::StatePtr closest = nullptr;
94  for (auto &kv : *contacts_) {
95  sc::Contact &cnt = kv.second;
96 
97  // Ignore same team
98  if (cnt.id().team_id() == std::get<0>(it->second).id().team_id()) {
99  continue;
100  }
101 
102  // Is the entity within the boundary?
103  if (std::get<1>(it->second)->contains(cnt.state()->pos())) {
104  if (closest == nullptr ||
105  (state_->pos() - closest->pos()).norm() < min_dist) {
106  closest = cnt.state();
107  min_dist = (state_->pos() - closest->pos()).norm();
108  }
109  }
110  }
111 
112  Eigen::Vector3d desired_point;
113  if (closest != nullptr) {
114  // Chase the entity
115  desired_point = closest->pos() + closest->vel() * time_->dt();
116  } else {
117  // If there are no entities to chase, move to center of boundary that
118  // we are trying to defend
119  desired_point = std::get<1>(it->second)->center();
120  }
121 
122  // Convert the desired point into a desired vector, relative to our own
123  // position.
