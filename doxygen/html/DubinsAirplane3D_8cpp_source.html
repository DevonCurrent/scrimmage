225 void DubinsAirplane3D::model(const vector_t &x , vector_t &dxdt , double t) {
226  dxdt[U] = 0;

234  double lambda = 1 - (pow(x[q0], 2) + pow(x[q1], 2) + pow(x[q2], 2) + pow(x[q3], 2));
235  dxdt[q0] = -0.5 * (x[q1]*x[P] + x[q2]*x[Q] + x[q3]*x[R]) + lambda * x[q0];
236  dxdt[q1] = +0.5 * (x[q0]*x[P] + x[q2]*x[R] - x[q3]*x[Q]) + lambda * x[q1];
237  dxdt[q2] = +0.5 * (x[q0]*x[Q] + x[q3]*x[P] - x[q1]*x[R]) + lambda * x[q2];
238  dxdt[q3] = +0.5 * (x[q0]*x[R] + x[q1]*x[Q] - x[q2]*x[P]) + lambda * x[q3];
239 
240  // Local position / velocity to global
241  // Normalize quaternion
242  sc::Quaternion quat(x[q0], x[q1], x[q2], x[q3]);
243  quat.normalize();
244 
245  quat = quat * quat_world_;
246  quat.normalize();
