168  std::string Print::formatTime(double time) {
169  if (std::isnan( time )) {
170  return "-----.---";
171  }
172  else {
173  std::ostringstream o;
174  o << std::internal << std::fixed << std::setprecision(3)

181  std::string Print::formatMsg(PrintEnums::WARN_LEVEL level, PrintData &data, std::string msg) {
182  std::ostringstream pre;
183  std::ostringstream formattedmsg;
184  std::string entitySection = "";
185 
186  // add entity_id if available
187  if (data.entity_id_ > 0) {
188  entitySection = "[Entity: " + std::to_string(data.entity_id_) + "]";
189  }
190 
191  // create our prefix
192  pre << formatTime(data.time_) << "s [" << data.name_ << "]" << entitySection << "[" << Print::getWarningLevel(level) << "]: ";
193 
194  // remove last character if it's a newline
195  if (msg.back() == '\n') {
196  msg.pop_back();
197  }
198 
199  // iterate and place the prefix at the beginning of each newline
200  formattedmsg << pre.str();
201  for (auto i = msg.cbegin() ; i != msg.cend() ; ++i) {
202  formattedmsg << *i;
203  // add prefix after each newline
204  if (*i == '\n') {
205  formattedmsg << pre.str();
206  }
207  }
208 
209  return formattedmsg.str();
210  }
