93 bool CaptureInBoundaryInteraction::step_entity_interaction(std::list<sc::EntityPtr> &ents,
94  double t, double dt) {
95  if (boundary_ == nullptr) {
96  return true;
97  }
98 
99  // Key : Possibly captured entity
100  // Value: Entity performing the capture
101  std::map<int, int> possible_captures;
102 
103  // Search over all entities whose team ID is the same as this boundary's
104  // team id and are within this boundary.
105  for (sc::EntityPtr ent : ents) {
106  // Determine if this entity has captured any other entities within the
107  // cool down period.
108  bool cool_down_expired = true;
109  auto it = prev_capture_times_.find(ent->id().id());
110  if (it != prev_capture_times_.end()) {
111  if (time_->t() <= (it->second + cool_down_period_)) {
112  cool_down_expired = false;
113  }
114  }
115 
116  if (cool_down_expired &&
117  ent->id().team_id() == boundary_shape_.id().team_id() &&
118  boundary_->contains(ent->state_truth()->pos())) {
119 
120  // Find all entities within capture range of this entity
121  std::vector<ID> rtree_neighbors;
122  parent_->rtree()->neighbors_in_range(ent->state_truth()->pos(),
123  rtree_neighbors,
124  capture_range_,
125  ent->id().id());
126 
127  // Only copy IDs whose team ID isn't the same as the boundary's
128  // team ID.
129  for (ID &id : rtree_neighbors) {
130  if (id.team_id() != boundary_shape_.id().team_id()) {
131  possible_captures[id.id()] = ent->id().id();
132  }
133  }
134  }
135  }
136 
137  // For all entities that are possibly captured, determine if they are
138  // within the boundary
139  for (sc::EntityPtr ent : ents) {
140  auto it = possible_captures.find(ent->id().id());
141  if (it != possible_captures.end()) {
142  // If the entity hasn't been captured yet and it's within the
143  // boundary, it is captured
144  if (already_captured_.count(it->first) == 0 &&
145  boundary_->contains(ent->state_truth()->pos())) {
146  ent->collision();
147 
148  // Keep track of all captured entities
149  already_captured_.insert(it->first);
150 
151  // Keep track of the capture time to limit the capture rate
152  prev_capture_times_[it->second] = time_->t();
153
