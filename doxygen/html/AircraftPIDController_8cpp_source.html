107 bool AircraftPIDController::step(double t, double dt) {
108  // Desired altitude to desired pitch
109  altitude_pid_.set_setpoint(vars_.input(desired_altitude_idx_));
110  double desired_pitch = -altitude_pid_.step(time_->dt(), state_->pos()(2));
111  desired_pitch = clamp(desired_pitch, -max_pitch_, max_pitch_);
112 
113  // Desired pitch to elevator

116  vars_.output(elevator_idx_, elevator);
117 
118  double desired_roll;
119  if (use_roll_control_) {

121  } else {
122  // Desired heading to desired roll
123  heading_pid_.set_setpoint(vars_.input(desired_heading_idx_));
124  desired_roll = -heading_pid_.step(time_->dt(), state_->quat().yaw());

126  desired_roll = clamp(desired_roll, -max_roll_, max_roll_);
127 
128  // Desired roll to aileron

132 
133  // Speed to throttle PID
