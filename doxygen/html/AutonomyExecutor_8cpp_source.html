114  std::string autonomy_name = "";

119  autonomy_name = str;
120  } else {
121  // Parse the autonomy parameters (e.g., param_name='value')
122  // Split the param_name and value, with equals sign in between
123  std::vector<std::string> tokens;
124  boost::split(tokens, str, boost::is_any_of("="));
125  if (tokens.size() == 2) {
126  // Remove the quotes from the value

128  std::remove_if(tokens[1].begin(),

133  tokens[1].end());

135  }

197 bool AutonomyExecutor::step_autonomy(double t, double dt) {
198  for (sc::AutonomyPtr autonomy : running_autonomies_) {
199  // cout << "Running... " << autonomy->name() << endl;
200  autonomy->shapes().clear();
201 
202  for (SubscriberBasePtr &sub : autonomy->subs()) {
203  for (auto msg : sub->pop_msgs<sc::MessageBase>()) {
204  sub->accept(msg);
205  }
206  }
207 
208  if (!autonomy->step_autonomy(time_->t(), time_->dt())) {
209  cout << "AutonomyExecutor: autonomy error- " << autonomy->name();
210  }
211 
212  if (show_shapes_) {
213  std::for_each(autonomy->shapes().begin(),
214  autonomy->shapes().end(), [&](auto &s) {
215  this->draw_shape(s);
216  });
217  }
218  }
219  return true;
220 }
