177 std::list<cv::Rect> MapGen2D::find_rectangles(cv::Mat &img, int threshold) {
178  // Make sure image is gray and apply threshold
179  cv::Mat gray;
180  cv::cvtColor(img, gray, cv::COLOR_BGRA2GRAY);
181 
182  cv::Mat thresh;
183  cv::threshold(gray, thresh, std::floor(threshold*255), 255,
184  cv::THRESH_BINARY_INV);
185 
186  cv::Mat img_rects = img.clone();
187 
188  std::list<cv::Rect> rects;
189 
190  cv::Mat I = thresh.clone();
191 
192  // Scan image looking for non-zero pixel
193  for (int i = 0; i < I.rows; i++) {
194  for (int j = 0; j < I.cols; j++) {
195  if (I.at<uchar>(i, j) > 0) {
196  // Non-zero pixel found, search for largest rectangle of
197  // non-zero pixels
198 
199  bool end_c_found = false;
200  int end_c = I.cols;
201  int end_r = I.rows;
202 
203  // Scan rows and columns looking for a zero-valued pixel
204  for (int r = i; r < end_r; r++) {
205  for (int c = j; c < end_c; c++) {
206  if (I.at<uchar>(r, c) == 0) {
207  // Zero pixel found
208  if (!end_c_found) {
209  // Use this column number as end column of
210  // rectangle
211  end_c_found = true;
212  end_c = c;
213  } else {
214  // Use this row number as end row of rectangle
215  end_r = r;
216  }
217  }
218  }
219 
220  if (end_c == I.cols) {
221  // If end_c is still the same as I.cols, then the
222  // rectangle extended to the end of the image.
223  end_c_found = true;
224  }
225  }
226 
227  cv::Rect rect(j, i, end_c-j, end_r-i);
228  cv::rectangle(img_rects, rect, cv::Scalar(0, 0, 255), 1, 8, 0);
229  cv::Mat roi = I(rect);
230  roi.setTo(0);
231 
232  rects.push_back(rect);
233  }
234  }
235  }
236 
237  if (enable_map_boundary_) {
238  rects.push_back(cv::Rect(0, 0, img.cols, 1)); // top rect
239  rects.push_back(cv::Rect(0, 0, 1, img.rows)); // left rect
240  rects.push_back(cv::Rect(img.cols, 0, 1, img.rows)); // right rect
241  rects.push_back(cv::Rect(0, img.rows, img.cols, 1)); // bottom rect
242  }
243 
244 
245  if (show_map_debug_) {
246  cout << "Number of rectangles: " << rects.size() << endl;
247  cv::imshow("Original", img);
248  cv::imshow("Gray", gray);
249  cv::imshow("Thresh", thresh);
250  cv::imshow("Rects", img_rects);
251  cv::waitKey(0);
252  }
253  return rects;
254 }
