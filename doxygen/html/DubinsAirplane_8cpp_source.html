103 bool DubinsAirplane::step(double t, double dt) {
104  // Get inputs and saturate
105  speed_ = boost::algorithm::clamp(vars_.input(desired_speed_idx_), speed_min_, speed_max_);
106  pitch_ = boost::algorithm::clamp(vars_.input(desired_pitch_idx_), pitch_min_, pitch_max_);
107  roll_ = boost::algorithm::clamp(vars_.input(desired_roll_idx_), roll_min_, roll_max_);
108 
109  x_[X] = state_->pos()(0);
110  x_[Y] = state_->pos()(1);
111  x_[Z] = state_->pos()(2);
112 
113  // Directly set the vehicle's roll and pitch
114  x_[ROLL] = roll_;
115  x_[PITCH] = pitch_;
116  x_[YAW] = state_->quat().yaw();
117 
118  double prev_x = x_[X];
119  double prev_y = x_[Y];
120  double prev_z = x_[Z];
121 
122  ode_step(dt);
123 
124  // Compute velocity (change in position)
125  state_->vel()(0) = (x_[X] - prev_x) / dt;
126  state_->vel()(1) = (x_[Y] - prev_y) / dt;
127  state_->vel()(2) = (x_[Z] - prev_z) / dt;
128 
129  // Update state position
130  state_->pos()(0) = x_[X];
131  state_->pos()(1) = x_[Y];
132  state_->pos()(2) = x_[Z];
133 
134  // Update state orientation
135  state_->quat().set(x_[ROLL], x_[PITCH], x_[YAW]);
136 
137  if (write_csv_) {
138  // Log state to CSV
139  csv_.append(sc::CSV::Pairs{
140  {"t", t},
141  {"x", state_->pos()(0)},
142  {"y", state_->pos()(1)},
143  {"z", state_->pos()(2)},
144  {"roll", state_->quat().roll()},
145  {"pitch", state_->quat().pitch()},
146  {"yaw", state_->quat().yaw()},
147  {"speed", speed_}});
148  }
149  return true;
150 }

152 void DubinsAirplane::model(const vector_t &x , vector_t &dxdt , double t) {
153  // The DubinsAirplane equations of motion are from the following paper:
154  // Owen, Mark, Randal W. Beard, and Timothy W. McLain. "Implementing dubins
155  // airplane paths on fixed-wing uavs." Handbook of unmanned aerial
156  // vehicles. Springer, Dordrecht, 2015. 1677-1701.
157 
158  double xy_speed = speed_ * cos(x[PITCH]);
159  dxdt[X] = xy_speed * cos(x[YAW]);
160  dxdt[Y] = xy_speed * sin(x[YAW]);
161  dxdt[Z] = -speed_ * sin(x[PITCH]);
162 
163  dxdt[ROLL] = 0;
164  dxdt[PITCH] = 0;
165 
166  if (std::abs(speed_) < std::numeric_limits<double>::epsilon()) {
167  dxdt[YAW] = 0;
168  } else {
169  dxdt[YAW] = -g_ / speed_ * tan(roll_);
170  }
171 }
