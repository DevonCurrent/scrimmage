47  // Construct a list of Waypoints from protobuf waypoint list
48  wp_list_.clear();
49  for (int i = 0; i < wp_list.waypoints().size(); i++) {
50  wp_list_.push_back(Waypoint(wp_list.waypoints(i)));
51  }
52 
53  // Filter the waypoint list to ensure that two waypoints in succession
54  // are not equal. Some planners generate duplicate waypoints.
55  auto wps_are_close = [] (auto &wp0, auto &wp1) {
56  auto numbers_are_close = [] (const double &a, const double &b) {
57  return std::abs(a - b) < std::numeric_limits<double>::epsilon();
58  };
59  return numbers_are_close(wp0.latitude(), wp1.latitude())
60  && numbers_are_close(wp0.longitude(), wp1.longitude())
61  && numbers_are_close(wp0.altitude(), wp1.altitude());
62  };
63  wp_list_.unique(wps_are_close);
64

66  // prev_wp_it_ = wp_list_.begin();
67  curr_wp_it_ = wp_list_.begin();
68  next_wp_it_ = next_waypoint(curr_wp_it_);
69 }

93 std::list<Waypoint>::iterator WaypointListProcessor::
94 next_waypoint(const std::list<Waypoint>::iterator &it_wp) {
95  std::list<Waypoint>::iterator it_next;
96  switch (mode_) {
97  case scrimmage_msgs::WaypointList::FOLLOW_ONCE:
98  it_next = std::next(it_wp);
99  break;
100 
101  case scrimmage_msgs::WaypointList::BACK_AND_FORTH:
102  it_next = returning_stage_ ? std::prev(it_wp): std::next(it_wp);
103 
104  if (it_next == wp_list_.begin()) {
105  returning_stage_ = false;
106  } else if (it_next == wp_list_.end()) {
107  it_next = it_wp;
108  returning_stage_ = true;
109  }
110  break;
111 
112  case scrimmage_msgs::WaypointList::LOITER:
113  // TODO : Not implemented yet
114  break;
115 
116  case scrimmage_msgs::WaypointList::LOOP:
117  // Continuous repeat of waypoints
118  it_next = std::next(curr_wp_it_);
119 
120  if (it_next == wp_list_.end()) {
121  it_next = wp_list_.begin();
122  }
123  break;
124 
125  default :
126  std::string msg = "Waypoint mode not defined yet.";
127  throw std::runtime_error(msg);
128  }
129  return it_next;
