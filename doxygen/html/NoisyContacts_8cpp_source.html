85 bool NoisyContacts::step() {
86  auto gener = parent_->random()->gener();
87  auto msg = std::make_shared<Message<ContactMap>>();
88 
89  // Create noisy versions of all contacts that aren't own vehicle
90  for (auto &kv: (*parent_->contacts())) {
91  // ignore own vehicle
92  if (kv.second.id().id() == parent_->id().id()) {
93  continue;
94  }
95 
96  // Construct a StateWithCovariance.
97  // Covar Matrix is 3x3 (x, y, z) with 5.0 along diagonal
98  std::shared_ptr<StateWithCovariance> state =
99  std::make_shared<StateWithCovariance>(*(kv.second.state()), 3, 3,
100  5.0);
101 
102  // Add noise to position and velocity
103  for (int i = 0; i < 3; i++) {
104  state->pos()(i) += (*pos_noise_[i])(*gener);
105  state->vel()(i) += (*vel_noise_[i])(*gener);
106  }
107 
108  // add noise in roll, pitch, yaw order
109  state->quat() = state->quat()
110  * Quaternion(Eigen::Vector3d::UnitX(), (*orient_noise_[0])(*gener))
111  * Quaternion(Eigen::Vector3d::UnitY(), (*orient_noise_[1])(*gener))
112  * Quaternion(Eigen::Vector3d::UnitZ(), (*orient_noise_[2])(*gener));
113 
114  // Construct a new Contact in place
115  msg->data.emplace(kv.second.id().id(), Contact(kv.second.id(),
116  std::static_pointer_cast<State>(state)));
117  }
118 
119  // Publish the noisy contact map
120  pub_->publish(msg);
121
