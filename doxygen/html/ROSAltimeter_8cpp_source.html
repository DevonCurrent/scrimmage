100 bool ROSAltimeter::step() {
101  // Obtain current state information
102  sc::StatePtr &state = parent_->state_truth();
103 
104  // Scrimmage is in East North Up (ENU)
105  // For you to get the ROS data in North East Down (NED): switch x and y, and negate z outside scrimmage.
106  // For more information on the MavROS message format view this page:
107  // https://mavlink.io/en/messages/common.html#ALTITUDE
108 
109  // Fill Altitude Message
110  mavros_msgs::Altitude alt_msg;
111  // Scrimmage is in cartesian, use Geographic lib to get lat, long, alt for monotonic
112  double lat, lon, alt;
113  // cartesian(x, y, z) to (lat, long, alt) using Geographic lib
114  parent_->projection()->Reverse(state->pos()(0), state->pos()(1), state->pos()(2), lat, lon, alt);
115  alt_msg.monotonic = monotonic_;
116  // If you print here monotonic will look like it's changing, but if you ROSTopic echo it will be a static number.
117  // cout << alt_msg.monotonic << endl;
118 
119  // amsl = above sea level (meters)
120  // get amsl from the same place as monotonic altitude, but this DOES change throughout the simulation
122  // This altitude measure is strictly above mean sea level and might be non-monotonic (it might reset on
123  // events like GPS lock or when a new QNH value is set). It should be the altitude to which global altitude
124  // waypoints are compared to. Note that it is *not* the GPS altitude, however, most GPS modules already
125  // output MSL by default and not the WGS84 altitude.
126  alt_msg.amsl = static_cast<float>(alt);
127 
128  // local = scrimmage z position of state
130  // This is the local altitude in the local coordinate frame. It is not the altitude above home, but in reference
131  // to the coordinate origin (0, 0, 0). It is up-positive.
132  alt_msg.local = state->pos()(2);
133 
134  // local - monotonic = relative
136  // This is the altitude above the home position. It resets on each change of the current home position.
137  alt_msg.relative = alt_msg.local - alt_msg.monotonic;
138 
139  // alt_terrain = amsl
141  // This is the altitude above terrain. It might be fed by a terrain database or an altimeter.
142  // Values smaller than -1000 should be interpreted as unknown.
143  alt_msg.terrain = alt_msg.amsl;
144 
145  // clearance = same as terrain // laser alt in scrimmage one day?
147  // This is not the altitude, but the clear space below the system according to the fused clearance estimate.
148  // It generally should max out at the maximum range of e.g. the laser altimeter. It is generally a moving target.
149  // A negative value indicates no measurement available.
150  alt_msg.bottom_clearance = alt_msg.amsl;
151 
152  // Header
153  std_msgs::Header header; // empty header
154  // TODO: header.frame = ? // No system for ROS Frame ID's yet
155  header.stamp = ros::Time::now(); // time
156  alt_msg.header = header;
157 
158  // Publish Altimeter information
159  altimeter_pub_.publish(alt_msg);
160 
161  return true;
162 }
