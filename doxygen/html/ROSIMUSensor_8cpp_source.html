114 Eigen::Vector3d ROSIMUSensor::gravity_ned_from_lla(Eigen::Vector3d lla) {
115  Eigen::Vector3d g;
116 
117  lla(0) = Angles::deg2rad(lla(0));
118  lla(1) = Angles::deg2rad(lla(1));
119 
120  double sLat = sin(lla(0));
121  double cLat = cos(lla(0));
122 
123  double N = earth_radius / sqrt(1.0 - ecc2 * sLat * sLat);
124  double x = (N + lla(2)) * cLat * cos(lla(1));
125  double y = (N + lla(2)) * cLat * sin(lla(1));
126  double z = ((b2 / a2) * N + lla(2)) * sLat;
127 
128  double x2 = x * x;
129  double y2 = y * y;
130  double z2 = z * z;
131  double r2 = x2 + y2 + z2 - e2;
132  double u2 = 0.5 * r2 * (1.0 + sqrt(1.0 + 4.0 * e2 * z2 / (r2 * r2)));
133  double u = sqrt(u2);
134  double v2 = u2 + e2;
135  double v = sqrt(v2);
136 
137  double Beta = atan2(z * v, u * sqrt(x2 + y2));
138  double sBeta = sin(Beta);
139  double sBeta2 = sBeta * sBeta;
140  double cBeta = cos(Beta);
141 
142  double w = sqrt(u2 + e2 * sBeta2) / v;
143  double q = 0.5 * ((1.0 + 3.0 * u2 / e2) * atan2(e, u) - 3.0 * u / e);
144  double q0 = 0.5 * ((1.0 + 3.0 * b2 / e2) * atan2(e, earth_semiminor_axis) - 3.0 * earth_semiminor_axis / e);
145  double qp = 3.0 * (1.0 + u2 / e2) * (1.0 - (u / e) * atan2(e, u)) - 1.0;
146 
147  // % //Mass Attraction Only
148  double Gu = -(wgs84_grav / v2 + omega2 * a2 * e * qp * (0.5 * sBeta2 - 1.0 / 6.0) / (q0 * v2)) / w;
149  double Gb = omega2 * sBeta * cBeta * a2 * q / (w * q0 * v);
150  double Psi = atan2(z, sqrt(x2 + y2));
151  double sPsi = sin(Psi);
152  double cPsi = cos(Psi);
153  double Alfa = lla(0) - Psi;
154  double sAlfa = sin(Alfa);
155  double cAlfa = cos(Alfa);
156  double Gr = Gu * (cPsi * cBeta * u / v + sPsi * sBeta) / w + Gb * (sPsi * cBeta * u / v - cPsi * sBeta) / w;
157  double Gp = -Gu * (sPsi * cBeta * u / v - cPsi * sBeta) / w + Gb * (cPsi * cBeta * u / v + sPsi * sBeta) / w;
158 
159  g(0) = -Gr * sAlfa + Gp * cAlfa;
160  g(1) = 0.0;
161  g(2) = -Gr * cAlfa - Gp * sAlfa;
162 
163  return g;
164 }

227 Eigen::Vector3d ROSIMUSensor::get_delta_theta(Eigen::Quaterniond qBodyToECEFt1Hat, Eigen::Quaterniond qBodyToECEFt2, double inertialDeltaT) {
228  Eigen::Matrix3d CBodyToECEFt1Hat(qBodyToECEFt1Hat);
229  Eigen::Matrix3d CBodyToECEFt2(qBodyToECEFt2);
230  Eigen::Matrix3d CBodyt2ToBodyt1 = CBodyToECEFt1Hat.transpose() * CBodyToECEFt2;
231 
232  Eigen::Matrix3d Ak = CBodyt2ToBodyt1;
233  Eigen::Vector3d dTHat = inv_skew_sym(CBodyt2ToBodyt1);
234  Eigen::Vector3d errVect;
235  for (int nn = 0; nn < 3; nn++) {
236  Eigen::Matrix3d AkHat = skew_sym(dTHat).exp();
237  errVect = inv_skew_sym(AkHat.transpose() * Ak);
238  dTHat = errVect + dTHat;
239  }
240  // % .0035 deg/Hr is the quoted gyro bias for the HG9900, a navigation grade
241  // % IMU
242  if (errVect.norm() >= 0.0035 * (M_PI / 180.0) * (1 / 3600.0) * (inertialDeltaT) * (1 / 100.0)) {
243  cout << "Warning: Check delta-theta calculations inside get_delta_theta!!!" << endl;
244  }
245  Eigen::Vector3d DeltaThetaBodyWRTECEFInBody = dTHat;
246 
247  Eigen::Vector3d EarthRateVector;
248  EarthRateVector << 0, 0, earth_rate;
249  Eigen::Vector3d deltaThetaBodyWRTInertialInBody = DeltaThetaBodyWRTECEFInBody + inertialDeltaT * CBodyToECEFt1Hat.transpose() * EarthRateVector;
250 
251  return deltaThetaBodyWRTInertialInBody;
252 }

263 Eigen::Quaterniond ROSIMUSensor::integrate_quaternion(Eigen::Quaterniond qBToA, Eigen::Vector3d deltaThetasBFrame) {
264  Eigen::Quaterniond QPredictor(qBToA.coeffs() + omega_to_q_dot(qBToA, deltaThetasBFrame).coeffs());
265  Eigen::Quaterniond QCorrector(qBToA.coeffs() + omega_to_q_dot(QPredictor, deltaThetasBFrame).coeffs());
266  // Perform the addition
267  Eigen::Quaterniond sum;
268  sum.w() = QPredictor.w() + QCorrector.w(); // Add the scalar portion
269  sum.vec() = QPredictor.vec() + QCorrector.vec(); // Add the vector portion
270  Eigen::Quaterniond qBToAUpdated(0.5 * sum.coeffs()); // (QPredictor + QCorrector);
271  qBToAUpdated.coeffs() = qBToAUpdated.coeffs() / qBToAUpdated.norm();
272  return qBToAUpdated;
273 }

285 Eigen::Vector3d ROSIMUSensor::get_deltaV(Eigen::Vector3d posECEF, Eigen::Vector3d velECEF, Eigen::Quaterniond qBodyToECEF, double InertialDeltaT) {
286  double k1 = -1, k2 = -1.25;
287 
288  Eigen::Vector3d Acct1 = (velECEF - vel_t1) / InertialDeltaT;
289  Eigen::Vector3d PosLLAt1 = ecef_to_lla(pos_ECEF_t1);
290  c_ECEF_To_NED_t1 = ecef_to_ned_rotation(PosLLAt1(0), PosLLAt1(1));
291  Eigen::Vector3d GravityNED = gravity_ned_from_lla(PosLLAt1);
292 
293  Eigen::Vector3d GravityECEF = c_ECEF_To_NED_t1.transpose() * GravityNED;
294  Eigen::Matrix3d CECEFToBodyTruth(prev_qBody_to_ECEF);
295  CECEFToBodyTruth.transposeInPlace();
296 
297  Eigen::Vector3d EarthRateVector;
298  EarthRateVector << 0, 0, earth_rate;
299  Eigen::Matrix3d EarthRateSkewSym = skew_sym(EarthRateVector);
300  Eigen::Vector3d deltaV = CECEFToBodyTruth * (Acct1 + 2.0 * EarthRateSkewSym * vel_t1 + EarthRateSkewSym * EarthRateSkewSym * pos_ECEF_t1 - GravityECEF);
301  deltaV = deltaV * InertialDeltaT;
302 
303  Eigen::Matrix3d CBodyToECEFt1Hat(qBody_to_ECEF_hat);
304  Eigen::Vector3d LLA = ecef_to_lla(p_hat);
305  GravityNED = gravity_ned_from_lla(LLA);
306 
307  Eigen::Matrix3d CECEFToNED = ecef_to_ned_rotation(LLA(0), LLA(1));
308  GravityECEF = CECEFToNED.transpose() * GravityNED;
309 
310  Eigen::Vector3d PErr = pos_ECEF_t1 - p_hat;
311  Eigen::Vector3d VErr = vel_t1 - v_hat;
312 
313  Eigen::Vector3d AAdd = -k1 * PErr - k2 * VErr;
314  deltaV = (deltaV + InertialDeltaT * CBodyToECEFt1Hat.transpose() * AAdd);
315 
316  Eigen::Vector3d Ahat = CBodyToECEFt1Hat * deltaV / InertialDeltaT - 2 * EarthRateSkewSym * v_hat - EarthRateSkewSym * EarthRateSkewSym * p_hat + GravityECEF;
317  Eigen::Vector3d tmpVhat = v_hat + (InertialDeltaT)*Ahat;
318  p_hat = p_hat + (InertialDeltaT)*v_hat + 0.5 * (InertialDeltaT * InertialDeltaT) * Ahat;
319  v_hat = tmpVhat;
320 
321  return deltaV;
322 }

324 bool ROSIMUSensor::step() {
325  // Obtain current state information
326  sc::StatePtr &state = parent_->state_truth();
327  double time_now = time_->t();
328  double dt = time_now - prev_time_;
329  prev_time_ = time_now;
330 
331  // Fill IMU Message
332  sensor_msgs::Imu imu_msg;
333 
334  // Header
335  std_msgs::Header header; // empty header
336  // TODO: header.frame = ? // No system for ROS Frame ID's yet
337  header.stamp = ros::Time::now(); // time
338  imu_msg.header = header;
339 
340  // get the geodetic origin specified in the mission file
341  MissionParsePtr mp_ = parent_->mp();
342  double lat = mp_->latitude_origin();
343  double lon = mp_->longitude_origin();
344  double alt = mp_->altitude_origin();
345 
346  // convert origin to ECEF and derive a rotation between ENU and ECEF coordinates
347  Eigen::Vector3d ecefOrigin = lla_to_ecef(lat, lon, alt);
348  Eigen::Matrix3d enuToEcef = enu_to_ecef_rotation(lat, lon);
349  Eigen::Quaterniond qEnuToEcef(enuToEcef);
350 
351  // convert the current position (ENU) to ECEF
352  Eigen::Vector3d ecefPos = state->pos();
353  ecefPos = enuToEcef * ecefPos; // rotate
354  ecefPos = ecefPos + ecefOrigin; // translate
355 
356  // convert (ENU) velocity to ECEF
357  Eigen::Vector3d ecefVel = state->vel();
358  ecefVel = enuToEcef * ecefVel; // rotate
359 
360  // calculate the body to ECEF orientation
361  Eigen::Quaterniond qbodyToEnu = state->quat();
362  Eigen::Quaterniond qbodyToECEF = qEnuToEcef * qbodyToEnu;
363 
364  // calculate angular velocity and linear acceleration based on this and the previous frames
365  if (!first_sample_collected) { // for the first sample, just store it, we need 2 samples to perform a calculation
366  pos_ECEF_t1 = ecefPos;
367  vel_t1 = ecefVel;
368  prev_qBody_to_ECEF = qbodyToECEF;
369  p_hat = pos_ECEF_t1;
370  v_hat = vel_t1;
371  qBody_to_ECEF_hat = qbodyToECEF;
372  first_sample_collected = true; // first sample has been collected
373  } else {
374  // calculate deltas
375  Eigen::Vector3d deltaV = get_deltaV(ecefPos, ecefVel, qbodyToECEF, dt);
376  Eigen::Vector3d deltaTheta = get_delta_theta(qBody_to_ECEF_hat, qbodyToECEF, dt);
377 
378  // update values for next frame
379  qBody_to_ECEF_hat = propagate_quaternion(qBody_to_ECEF_hat, deltaTheta, dt);
380  pos_ECEF_t1 = ecefPos;
381  vel_t1 = ecefVel;
382  prev_qBody_to_ECEF = qbodyToECEF;
383 
384  // Populate the message with orientation
385  imu_msg.orientation.x = qbodyToECEF.x();
386  imu_msg.orientation.y = qbodyToECEF.y();
387  imu_msg.orientation.z = qbodyToECEF.z();
388  imu_msg.orientation.w = qbodyToECEF.w();
389 
390  if (error_sim == nullptr) {
391  // Angular Velocity
392  imu_msg.angular_velocity.x = deltaTheta.x();
393  imu_msg.angular_velocity.y = deltaTheta.y();
394  imu_msg.angular_velocity.z = deltaTheta.z();
395 
396  // Linear Acceleration
397  imu_msg.linear_acceleration.x = deltaV.x();
398  imu_msg.linear_acceleration.y = deltaV.y();
399  imu_msg.linear_acceleration.z = deltaV.z();
400  } else {
401  // add error to the deltaV and deltaTheta values
402  NoisyIMUData noisyData = error_sim->EachCycle(error_budget, deltaV, deltaTheta);
403 
404  // Angular Velocity
405  imu_msg.angular_velocity.x = noisyData.noisyDeltaTheta.x();
406  imu_msg.angular_velocity.y = noisyData.noisyDeltaTheta.y();
407  imu_msg.angular_velocity.z = noisyData.noisyDeltaTheta.z();
408 
409  // Linear Acceleration
410  imu_msg.linear_acceleration.x = noisyData.noisyDeltaV.x();
411  imu_msg.linear_acceleration.y = noisyData.noisyDeltaV.y();
412  imu_msg.linear_acceleration.z = noisyData.noisyDeltaV.z();
413  }
414 
415  // Publish IMU information
416  imu_pub_.publish(imu_msg);
417 
418  // Write IMU data to CSV
419  // Write the CSV file to the root log directory file name = imu_data.csv
420  if (!csv.output_is_open()) {
421  cout << "File isn't open. Can't append to CSV" << endl;
422  }
423  csv.append(sc::CSV::Pairs{
424  {"time", time_now},
425  {"dt", dt},
426  {"ECEF_POSX", ecefPos.x()},
427  {"ECEF_POSY", ecefPos.y()},
428  {"ECEF_POSZ", ecefPos.z()},
429  {"ECEF_VELX", ecefVel.x()},
430  {"ECEF_VELY", ecefVel.y()},
431  {"ECEF_VELZ", ecefVel.z()},
432  {"bodyToEcef_X", qbodyToECEF.x()},
433  {"bodyToEcef_Y", qbodyToECEF.y()},
434  {"bodyToEcef_Z", qbodyToECEF.z()},
435  {"bodyToEcef_W", qbodyToECEF.w()},
436  {"dV_X", deltaV.x()},
437  {"dV_Y", deltaV.y()},
438  {"dV_Z", deltaV.z()},
439  {"dTheta_X", deltaTheta.x()},
