62 bool Interface::init_network(Interface::Mode_t mode, const std::string &ip, int port) {
63  mode_ = mode;
64  ip_ = ip;
65  port_ = port;
66 
67  if (mode_ == server) {
68 #if ENABLE_GRPC == 1
69  std::string result = ip_ + ":" + std::to_string(port_);
70  ScrimmageServiceImpl frame_service(this);
71  grpc::ServerBuilder builder;
72  builder.AddListeningPort(result, grpc::InsecureServerCredentials());
73  builder.RegisterService(&frame_service);
74  server_ = builder.BuildAndStart();
75  std::cout << "Server listening on " << result << std::endl;
76 
77  // start thread with shutdown option
78  // https://stackoverflow.com/a/36182429
79  auto serving_thread = std::thread(&Interface::start_server, this);
80  auto future = frame_service.exit_requested.get_future();
81  future.wait();
82  server_->Shutdown();
83  serving_thread.join();
84 #else
85  cout << "WARNING: GRPC DISABLED!" << endl;
86 #endif
87  } else if (mode_ == client) {
88 #if ENABLE_GRPC
89  std::string result = ip_ + ":" + std::to_string(port_);
90  std::shared_ptr<Channel> channel(
91  grpc::CreateChannel(result, grpc::InsecureChannelCredentials())
92  );
93  std::unique_ptr<scrimmage_proto::ScrimmageService::Stub> frame_temp(scrimmage_proto::ScrimmageService::NewStub(channel));
94  scrimmage_stub_ = std::move(frame_temp);
95  cout << "Client connecting to " << result << endl;
96 #else
97  cout << "WARNING: GRPC DISABLED!" << endl;
98 #endif
99  }
100  return true;
101 }
