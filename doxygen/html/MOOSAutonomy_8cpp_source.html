90  MOOSNode::NodeReportType type;

96

104  s->pos()(0), s->pos()(1),
105  s->vel().norm(), heading,

108  }

112 
113  // Convert heading to local cartesian
114  angles_from_moos_.set_angle(ang::rad2deg(s.quat().yaw()));
115  s.quat().set(s.quat().roll(), s.quat().pitch(),
116  ang::deg2rad(angles_from_moos_.angle()));
117

120  desired_state_->pos() = s.pos();
