198 bool ScrimmageOpenAIAutonomy::send_env() {
199  sp::Environment env;
200  // Setup action space parameters
201  auto discrete_space = env.mutable_action_spaces()->mutable_discrete();
202  *(discrete_space) = {action_space.discrete_count.begin(), action_space.discrete_count.end()};
203  for (std::pair<double, double> cont_extrema : action_space.continuous_extrema) {
204  sp::ContExtrema * proto_extrema = env.mutable_action_spaces()->add_continuous();
205  proto_extrema->set_min(cont_extrema.first);
206  proto_extrema->set_max(cont_extrema.second);
207  }
208  // Setup observation space parameters
209  // This for loop should only occur once as the agent should only know about its own sensors
210  for (auto agent : observations_.ext_sensor_vec()) {
211  // For every sensor the agent has
212  for (auto sensor : agent) {
213  auto obs_space = sensor->observation_space;
214  // Add discrete observation spaces to the protobuf environment
215  for (int discrete_obs : obs_space.discrete_count) {
216  env.mutable_observation_spaces()->add_discrete(discrete_obs);
217  }
218  // Add continuous/ observation spaces to the protobuf environment
219  for (std::pair<double, double> cont_extrema : obs_space.continuous_extrema) {
220  sp::ContExtrema * proto_extrema = env.mutable_observation_spaces()->add_continuous();
221  proto_extrema->set_min(cont_extrema.first);
222  proto_extrema->set_max(cont_extrema.second);
223  }
224  }
225  }
226  // Send over params through GRPC
227  std::vector<std::string> necessary_params = {"logdir", "hyperparams_file"};
228  auto &proto_map = *env.mutable_params();
229  for (std::string param_key : necessary_params) {
230  auto search = params_.find(param_key);
231  if (search != params_.end()) {
232  proto_map[param_key] = search->second;
233  }
234  }
235 
236  if (!openai_stub_) return false;
237  sp::Empty reply;
238 
239  grpc::ClientContext context;
240  auto deadline = std::chrono::system_clock::now() + std::chrono::seconds(2);
241  context.set_deadline(deadline);
242  grpc::Status status = openai_stub_->SendEnvironment(&context, env, &reply);
243  if (!status.ok()) {
244  std::cout << "Can't Connect: message " << status.error_code()
245  << ", " << status.error_message() << std::endl;
246  }
247  return status.ok();
248 }
