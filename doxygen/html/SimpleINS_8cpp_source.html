96 bool SimpleINS::step() {

99  msg_SS->data = sea_state_gps_;

104 
105  auto msg = std::make_shared<Message<StateWithCovariance>>();
106

111 
112 
113  // Acceleration

115  vel_Nminus1 = ns.vel();
116  for (int i = 0; i < 3; i++) {
117  accel_(i) = abs(accel_(i) / .01);
118  if (accel_(i) > 1) {
119  accel_(i) = 1;
120  }
121  }
122 
123 
124  // Use gen in order to create a growing covariance
125  double pos_noise_0 = (*pos_noise_[0])(*gener) * (accel_(0));
126  double pos_noise_1 = (*pos_noise_[1])(*gener) * (accel_(1));
127  const int arrayNum[4] = {15, 30, 45, 60};
128 
129  if (!gps_fix_) {
130  m_(0, 0) += (pos_noise_0 < 0 ? -1*pos_noise_0 : pos_noise_0);
131  m_(1, 1) += (pos_noise_1 < 0 ? -1*pos_noise_1 : pos_noise_1);
132 
133  prev_time_ = time_->t();
134  int randIdx = rand() % 4;
135  surface_timer_ = arrayNum[randIdx];
136  if (sea_state_gps_ > 1) {
137  surface_timer_ += 60;
138  }
139  } else {
140  // GPS fix on - wait on surface timer and snap back to an identity matrix
141  m_(0, 0) += (pos_noise_0 < 0 ? -1*pos_noise_0 : pos_noise_0);
142  m_(1, 1) += (pos_noise_1 < 0 ? -1*pos_noise_1 : pos_noise_1);
143  // GPS fix on - wait on surface timer and snap back to an identity matrix
144  if (time_->t() - prev_time_ > surface_timer_) {
145  m_ = Eigen::MatrixXd::Identity(ns.covariance().rows(), ns.covariance().cols());
146  // Reset the position error accumulator
147  pos_error_accum_ = Eigen::Vector3d::Zero();
148  }
149  }
150 
151 
152 
153  // Create noisy position / velocity
154  for (int i = 0; i < 3; i++) {
155  // std::cout << "Accelration for " << i << ", is: " << (accel_(i)) << std::endl;
156  pos_error_accum_(i) += (*pos_noise_[i])(*gener) * (accel_(i));
157  // msg->data.pos()(i) = ns.pos()(i) + (*pos_noise_[i])(*gener);
158  msg->data.vel()(i) = ns.vel()(i) + (*vel_noise_[i])(*gener);
159  }
160  msg->data.pos() = ns.pos() + pos_error_accum_;
161 
162  msg->data.quat() = ns.quat()
163  * Quaternion(Eigen::Vector3d::UnitX(), (*orient_noise_[0])(*gener))
164  * Quaternion(Eigen::Vector3d::UnitY(), (*orient_noise_[1])(*gener))
165  * Quaternion(Eigen::Vector3d::UnitZ(), (*orient_noise_[2])(*gener));
166 
167  msg->data.set_covariance(m_);

169  double sig_L;
170  double sig_S;
171 
172  if (m_(0, 0) < m_(1, 1)) {
173  sig_L = sqrt(m_(1, 1));
174  sig_S = sqrt(m_(0, 0));
175  } else {
176  sig_L = sqrt(m_(0, 0));
177  sig_S = sqrt(m_(1, 1));
178  }
179  double sig_w = sig_S / sig_L;
180  double CEP;
181  if (sig_w < 0.5) {
182  CEP = sig_L * (0.67 + 0.8 * sig_w * sig_w);
183  } else {
184  CEP = 0.59 * sig_L * (1 + sig_w);
185  }
186  auto msg_cep = std::make_shared<sc::Message<double>>();

188  pub_cep_->publish(msg_cep);
189  // Publish StateWCovariance msg
190  pub_->publish(msg);
191 
192  // Update the entity's state.
193  *(parent_->state()) = static_cast<sc::State>(msg->data);
194 
195  return true;
