258 bool get_vec_of_vecs(const std::string &str,
259  std::vector<std::vector<std::string>> &out,
260  const std::string &delims) {
261  // Of the form...
262  // [a b c d]
263  // [e f g h]
264  // [h i j k]
265  out.clear();
266 
267  std::string str_copy(str);
268  remove_leading_spaces(str_copy);
269  remove_trailing_spaces(str_copy);
270 
271  // First split based on user defined delimiters (typically ", ")
272  std::vector<std::string> tokens;
273  boost::algorithm::split(tokens, str_copy, boost::is_any_of(delims),
274  boost::token_compress_on);
275 
276  // The tokens vector consists of the delimited items. Break up each vector
277  // based on the start "[" and stop "]" delimiters. However, we will allow
278  // one-level of nested start and stop delimiters.
279  std::vector<std::string> items;
280  int open_bracket_count = 0;
281  std::string item = "";
282  for (std::string token : tokens) {
283 
284  // Remove whitespace
285  token.erase(
286  std::remove_if(token.begin(),
287  token.end(),
288  [](unsigned char x){return std::isspace(x);}),
289  token.end());
290 
291  // cout << token << endl;
292 
293  for (unsigned int i = 0; i < token.size(); i++) {
294  if (token[i] == '[') {
295  if (open_bracket_count > 0) {
296  item += "[";
297  }
298  ++open_bracket_count;
299  } else if (token[i] == ']') {
300  if (open_bracket_count > 1) {
301  item += "]";
302  }
303  --open_bracket_count;
304 
305  if (open_bracket_count == 0) {
306  items.push_back(item);
307  out.push_back(items);
308  items.clear();
309  item = "";
310  }
311  } else if (open_bracket_count == 2) {
312  item += std::string(1, token[i]);
313  } else if (open_bracket_count > 0) {
314  item += std::string(1, token[i]);
315  }
316  }
317 
318  if (open_bracket_count == 1) {
319  items.push_back(item);
320  item.clear();
321  } else if (open_bracket_count == 0) {
322  out.push_back(items);
323  item.clear();
324  items.clear();
325  }
326  }
327 
328  // Remove any empty strings from the individual vectors
329  for (auto &vec : out) {
330  vec.erase(std::remove_if(vec.begin(), vec.end(), [] (auto &str) {
331  return str.empty();
332  }), vec.end());
333  }
334 
335  // Remove any empty vectors from the out vector of vectors
336  out.erase(std::remove_if(out.begin(), out.end(), [] (auto &vec) {
337  return vec.empty();
338  }), out.end());
339 
340  return true;
341 }
