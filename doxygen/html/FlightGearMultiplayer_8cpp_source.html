141 bool FlightGearMultiplayer::step_autonomy(double t, double dt) {
142  // Create the flight gear position msg
143  pos_msg_.time = XDR_encode64<float>(time_->t());
144  pos_msg_.lag = XDR_encode64<float>(time_->t());
145 
146  // Compute ECEF coordinates from x/y/z lat/lon/alt
147  double lat, lon, alt;
148  parent_->projection()->Reverse(state_->pos()(0), state_->pos()(1),
149  state_->pos()(2),
150  lat, lon, alt);
151  double ECEF_X, ECEF_Y, ECEF_Z;
152  earth_->Forward(lat, lon, alt, ECEF_X, ECEF_Y, ECEF_Z);
153 
154  pos_msg_.position[0] = XDR_encode64<double>(ECEF_X);
155  pos_msg_.position[1] = XDR_encode64<double>(ECEF_Y);
156  pos_msg_.position[2] = XDR_encode64<double>(ECEF_Z);
157 
158  // Convert local heading to GPS heading
159  angles_to_jsbsim_.set_angle(sc::Angles::rad2deg(state_->quat().yaw()));
160  sc::Quaternion local_quat(state_->quat().roll(), state_->quat().pitch(),
161  sc::Angles::deg2rad(angles_to_jsbsim_.angle()));
162 
163  // Convert local quaternion to ECEF coordinate
164  scrimmage::Quaternion rot = this->fromLonLatRad(sc::Angles::deg2rad(lon),
165  sc::Angles::deg2rad(lat));
166 
167  // The FlightGear Multiplayer packet expects the orientation to be in
168  // angle-axis form, where the axis magnitude is the rotation angle.
169  Eigen::AngleAxisd ang_axis(rot * local_quat);
170  Eigen::Vector3d ang_axis_mod = ang_axis.axis().normalized() * ang_axis.angle();
171 
172  pos_msg_.orientation[0] = XDR_encode<float>(ang_axis_mod(0));
173  pos_msg_.orientation[1] = XDR_encode<float>(ang_axis_mod(1));
174  pos_msg_.orientation[2] = XDR_encode<float>(ang_axis_mod(2));
175 
176  // Copy header and position data structs into memory buffer
177  char * msg = new char[msg_size_];
178  std::memcpy(msg, &header_msg_, sizeof(T_MsgHdr));
179  std::memcpy(msg+sizeof(T_MsgHdr), &pos_msg_, sizeof(T_PositionMsg));
180 
181  data_socket_->sendto(msg, msg_size_, 0, &net_address_);
182  delete[] msg;
183 
184  return true;
185 }
