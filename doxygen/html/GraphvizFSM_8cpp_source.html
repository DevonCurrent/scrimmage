130 bool GraphvizFSM::step_autonomy(double t, double dt) {
131  // If the current state only has one out edge (degree == 1) and doesn't
132  // have a trigger label to its next state, immediately move to that next
133  // state since this is just a designator for the "initial" state.
134  if (boost::out_degree(current_state_, fsm_graph_) == 1) {
135  typename boost::graph_traits<graph_t>::out_edge_iterator ei, ei_end;
136  for (boost::tie(ei, ei_end) = boost::out_edges(current_state_, fsm_graph_); ei != ei_end; ++ei) {
137  if (label_(*ei) == std::string("")) {
138  this->update_state_info(current_state_,
139  boost::target(*ei, fsm_graph_));
140  }
141  }
142  }
143  return true;
144 }
