100 bool WaypointDispatcher::step_autonomy(double t, double dt) {
101  if (wp_list_.waypoints().size() == 0) {
102  return true;
103  }
104 
105  Waypoint curr_wp_lla = *wp_it_;
106 
107  Eigen::Vector3d wp;
108  parent_->projection()->Forward(curr_wp_lla.latitude(),
109  curr_wp_lla.longitude(),
110  curr_wp_lla.altitude(), wp(0), wp(1), wp(2));
111 
112  Eigen::Vector3d prev_wp;
113  if (prev_wp_it_ == wp_it_) {
114  prev_wp = state_->pos();
115  } else {
116  Waypoint prev_wp_lla = *prev_wp_it_;
117  parent_->projection()->Forward(prev_wp_lla.latitude(), prev_wp_lla.longitude(),
118  prev_wp_lla.altitude(), prev_wp(0),
119  prev_wp(1), prev_wp(2));
120  }
121 
122  Eigen::Vector3d wp_line = wp - prev_wp;
123  Eigen::Vector3d wp_to_own_line = state_->pos() - prev_wp;
124 
125  // Project wp_to_own_line onto line between waypoints (wp_line)
126  double dot_prod = wp_line.dot(wp_to_own_line);
127  Eigen::Vector3d proj = dot_prod / pow(wp_line.norm(), 1) *
128  wp_line.normalized();
129 
130  Eigen::Vector3d track_point = prev_wp + proj + wp_line.normalized() * lead_distance_;
131 
132  if (dot_prod < 0) {
133  track_point = prev_wp + wp_line.normalized() * (lead_distance_);
134  }
135 
136  if ((track_point - prev_wp).norm() >= (wp - prev_wp).norm()) {
137  track_point = wp;
138  }
139

141  sphere->set_persistent(true);

153  }

164

168  (state_->pos() - wp).norm();
169  if (dist <= curr_wp_lla.position_tolerance()) {
170

172  switch (wp_list_.mode()) {
173  case WaypointList::WaypointMode::follow_once :
174  prev_wp_it_ = wp_it_;
175  ++wp_it_;
176 
177  if (wp_it_ == wp_list_.waypoints().end()) {
178  wp_it_ = prev_wp_it_;
179  done = true;
180  }
181  break;
182 
183  case WaypointList::WaypointMode::back_and_forth :
184  prev_wp_it_ = wp_it_;
185  wp_it_ = returning_stage_ ? --wp_it_ : ++wp_it_;
186 
187  if (wp_it_ == wp_list_.waypoints().begin()) {
188  returning_stage_ = false;
189  } else if (wp_it_ == wp_list_.waypoints().end()) {
190  wp_it_ = prev_wp_it_;
191  returning_stage_ = true;
192  }
193  break;
194 
195  case WaypointList::WaypointMode::loiter :
196  // TODO : Not implemented yet
197  break;
198 
199  case WaypointList::WaypointMode::racetrack :
200  // Continuous repeat of waypoints
201  prev_wp_it_ = wp_it_;
202  ++wp_it_;
203 
204  if (wp_it_ == wp_list_.waypoints().end()) {
205  wp_it_ = wp_list_.waypoints().begin();
206  }
207  break;
208

211  throw std::runtime_error(msg);

215  size_t idx = std::distance(wp_list_.waypoints().begin(), wp_it_);
