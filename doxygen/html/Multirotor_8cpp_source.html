298 void Multirotor::model(const vector_t &x , vector_t &dxdt , double t) {
299  // Omega values for each rotor
300  Eigen::VectorXd omega = ctrl_u_;
301  Eigen::VectorXd omega_sq = omega.cwiseProduct(omega);
302 
303  // Calculate force from combined rotor thrust
304  Eigen::Vector3d F_thrust(0, 0, 0);
305  Eigen::VectorXd rotor_thrust(rotors_.size());
306  Eigen::VectorXd rotor_torque(rotors_.size());
307 
308  for (unsigned int i = 0; i < rotors_.size(); i++) {
309  rotor_thrust(i) = c_T_ * omega_sq(i);
310  if (std::abs(omega[i] - wmin_) < std::numeric_limits<double>::epsilon()) { // if minimum throttle, servo is off
311  rotor_thrust[i] = 0.0;
312  }
313  rotor_torque(i) = rotors_[i].direction() * c_Q_ * omega_sq(i);
314  F_thrust(2) += rotor_thrust(i);
315  }
316 
317  // Calculate force from weight in body frame:
318  Eigen::Vector3d gravity_vector(0, 0, -mass_*g_);
319  Eigen::Vector3d F_weight = state_->quat().rotate_reverse(gravity_vector);
320 
321  // Calculate force from drag (TODO: Check source)
322  Eigen::Vector3d vel_body(x_[U], x_[V], x_[W]);
323  double vel_mag = vel_body.norm();
324  Eigen::Vector3d F_drag = vel_body * (-0.5 * c_D_ * vel_mag);
325 
326  // Calculate normal force
327  force_ext_body_[2] = force_ext_body_[2]-F_thrust[2];
328  if (force_ext_body_[2] > 0) { // only true if contacting ground
329  force_ext_body_[2] = -1.0*(F_weight[2]) - F_thrust[2]; // correctly calculate normal force
330  }
331 
332  if (force_ext_body_[2] < 0) { // normal force only pushes up from the ground
333  force_ext_body_[2] = 0.0;
334  }
335 
336  // Calculate total force
337  Eigen::Vector3d F_total = F_thrust + F_weight + F_drag + force_ext_body_;
338 
339  dxdt[U] = x[V]*x[R] - x[W]*x[Q] + F_total(0) / mass_;
340  dxdt[V] = x[W]*x[P] - x[U]*x[R] + F_total(1) / mass_;
341  dxdt[W] = x[U]*x[Q] - x[V]*x[P] + F_total(2) / mass_;
342 
343  // Calculate moments from thrust
344  Eigen::Vector3d Moments_thrust(0, 0, 0); // L, M, N
345 
346  for (unsigned int i = 0; i < rotors_.size(); i++) {
347  double length = rotors_[i].offset_length();
348  double xy_angle = rotors_[i].xy_angle();
349 
350  Moments_thrust(0) += rotor_thrust(i) * length * sin(xy_angle);
351  Moments_thrust(1) -= rotor_thrust(i) * length * cos(xy_angle);
352  Moments_thrust(2) += rotor_torque(i);
353  }
354 
355  Eigen::Vector3d Moments_total = Moments_thrust;
356 
357  Eigen::Vector3d pqr(x_[P], x_[Q], x_[R]);
358  Eigen::Vector3d pqr_dot = I_inv_ * (Moments_total - pqr.cross(I_*pqr));
359  dxdt[P] = pqr_dot(0);
360  dxdt[Q] = pqr_dot(1);
361  dxdt[R] = pqr_dot(2);
362 
363  // Update quaternion from angular accelerations
364  double lambda = 1 - (pow(x[q0], 2) + pow(x[q1], 2) + pow(x[q2], 2) + pow(x[q3], 2));
365  dxdt[q0] = -0.5 * (x[q1]*x[P] + x[q2]*x[Q] + x[q3]*x[R]) + lambda * x[q0];
366  dxdt[q1] = +0.5 * (x[q0]*x[P] + x[q2]*x[R] - x[q3]*x[Q]) + lambda * x[q1];
367  dxdt[q2] = +0.5 * (x[q0]*x[Q] + x[q3]*x[P] - x[q1]*x[R]) + lambda * x[q2];
368  dxdt[q3] = +0.5 * (x[q0]*x[R] + x[q1]*x[Q] - x[q2]*x[P]) + lambda * x[q3];
369 
370  // Local position / velocity to global
371  // Normalize quaternion
372  sc::Quaternion quat(x[q0], x[q1], x[q2], x[q3]);
373  quat.w() = x[q0];
374  quat.x() = x[q1];
375  quat.y() = x[q2];
376  quat.z() = x[q3];
377  quat.normalize();
378 
379  Eigen::Vector3d vel_local(x[U], x[V], x[W]);
380  Eigen::Vector3d vel_world = quat.rotate(vel_local);
381  dxdt[Xw] = vel_world(0);
382  dxdt[Yw] = vel_world(1);
383  dxdt[Zw] = vel_world(2);
384 
386 
387  Eigen::Vector3d acc_local = F_total / mass_; // this is in body frame
388  Eigen::Vector3d acc_world = quat.rotate(acc_local);
389  dxdt[Uw] = acc_world(0);
390  dxdt[Vw] = acc_world(1);
391  dxdt[Ww] = acc_world(2);
392 }
