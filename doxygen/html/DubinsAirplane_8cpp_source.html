152 void DubinsAirplane::model(const vector_t &x , vector_t &dxdt , double t) {
153  // The DubinsAirplane equations of motion are from the following paper:
154  // Owen, Mark, Randal W. Beard, and Timothy W. McLain. "Implementing dubins
155  // airplane paths on fixed-wing uavs." Handbook of unmanned aerial
156  // vehicles. Springer, Dordrecht, 2015. 1677-1701.
157 
158  double xy_speed = speed_ * cos(x[PITCH]);
159  dxdt[X] = xy_speed * cos(x[YAW]);
160  dxdt[Y] = xy_speed * sin(x[YAW]);
161  dxdt[Z] = -speed_ * sin(x[PITCH]);
162 
163  dxdt[ROLL] = 0;
164  dxdt[PITCH] = 0;
165 
166  if (std::abs(speed_) < std::numeric_limits<double>::epsilon()) {
167  dxdt[YAW] = 0;
168  } else {
169  dxdt[YAW] = -g_ / speed_ * tan(roll_);
170  }
171 }
