216 bool Straight::step_autonomy(double t, double dt) {
217  if (gen_ents_) {
218  if (time_->t() > (prev_gen_time_ + 2.0)) {
219  prev_gen_time_ = time_->t();
220 
221  // Create a state for the new entity
222  State s(*state_);
223 
224  // Rotate the heading of the new entity by 90 degrees and offset
225  // the initial position to the left of our current position to
226  // avoid collisions.
227  Eigen::AngleAxisd rot_90_z(M_PI/2.0, Eigen::Vector3d::UnitZ());
228  s.pos() = s.pos() + s.quat() * rot_90_z * (Eigen::Vector3d::UnitX() * 10);
229  s.quat() = rot_90_z * s.quat();
230 
231  // Create the GenerateEntity message
232  auto msg = std::make_shared<Message<scrimmage_msgs::GenerateEntity>>();
233  sc::set(msg->data.mutable_state(), s); // Copy the new state
234 
235  // The entity_tag must match the "tag" XML attribute of the entity
236  // to be generated in the mission file.
237  msg->data.set_entity_tag("gen_straight");
238 
239  // Modify the entity's color
240  auto kv_color = msg->data.add_entity_param();
241  kv_color->set_key("color");
242  kv_color->set_value("255, 255, 0");
243 
244  // Modify the entity's visual model
245  auto kv_visual = msg->data.add_entity_param();
246  kv_visual->set_key("visual_model");
247  kv_visual->set_value("sphere");
248 
249  pub_gen_ents_->publish(msg); // Publish the GenerateEntity message
250  }
251  }
252 
253  if (show_text_label_) {
254  // An example of changing a shape's property
255  if (t > 1.0 && t < (1.0 + dt)) {
256  text_shape_->set_opacity(0.5);
257  text_shape_->mutable_text()->set_text("Goodbye, SCRIMMAGE!");
258  draw_shape(text_shape_);
259  }
260 
261  // An example of removing a shape
262  if (t > 5.0 && t < (5.0 + dt)) {
263  text_shape_->set_persistent(false);
264  draw_shape(text_shape_);
265  }
266  }
267 
268  // Read data from sensors...
269  if (!noisy_state_set_) {
270  noisy_state_ = *state_;
271  }
272 
273  if (boundary_ != nullptr && enable_boundary_control_) {
274  if (!boundary_->contains(noisy_state_.pos())) {
275  // Project goal through center of boundary
276  Eigen::Vector3d center = boundary_->center();
277  center(2) = noisy_state_.pos()(2); // maintain altitude
278  Eigen::Vector3d diff = center - noisy_state_.pos();
279  goal_ = noisy_state_.pos() + diff.normalized() * 1e6;
280  }
281  }
282 
283  Eigen::Vector3d diff = goal_ - noisy_state_.pos();
284  Eigen::Vector3d v = speed_ * diff.normalized();
285 
287  // Convert desired velocity to desired speed, heading, and pitch controls
289  double heading = Angles::angle_2pi(atan2(v(1), v(0)));
290  vars_.output(desired_alt_idx_, goal_(2));
291  vars_.output(desired_speed_idx_, v.norm());
292  vars_.output(desired_heading_idx_, heading);
293 
294  noisy_state_set_ = false;
295  return true;
296 }
