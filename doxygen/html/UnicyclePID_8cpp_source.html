98 bool UnicyclePID::step(double t, double dt) {
99  heading_pid_.set_setpoint(vars_.input(desired_heading_idx_));
100  vars_.output(turn_rate_idx_, heading_pid_.step(time_->dt(), state_->quat().yaw()));
101 
102  // Reconstruct original velocity vector (close, but not exact
103  double x_vel = vars_.input(desired_speed_idx_) / sqrt(1 + pow(tan(vars_.input(desired_heading_idx_)), 2));
104  double y_vel = x_vel * tan(vars_.input(desired_heading_idx_));
105  Eigen::Vector3d vel(x_vel, y_vel, vars_.input(desired_alt_idx_) - state_->pos()(2));
106  vel = vel.normalized() * vars_.input(desired_speed_idx_);
107 
108  // Track desired pitch
109  double desired_pitch = atan2(vel(2), vel.head<2>().norm());
110  pitch_pid_.set_setpoint(desired_pitch);
111  vars_.output(pitch_rate_idx_, -pitch_pid_.step(time_->dt(), - state_->quat().pitch()));
112 
113  // Track zero roll
114  roll_pid_.set_setpoint(0);
115  vars_.output(roll_rate_idx_, -roll_pid_.step(time_->dt(), - state_->quat().roll()));
116 
117  if (show_shapes_) {
118  line_shape_->set_persistent(true);
119  sc::set(line_shape_->mutable_color(), 255, 0, 0);
120  line_shape_->set_opacity(1.0);
121  sc::set(line_shape_->mutable_line()->mutable_start(), state_->pos());
122  sc::set(line_shape_->mutable_line()->mutable_end(), vel + state_->pos());
123  draw_shape(line_shape_);
124  }
125 
126  // Only use PID if controlling acceleration, otherwise, we directly set the
127  // forward speed
128  double desired_speed = vars_.input(desired_speed_idx_);
129  if (use_accel_) {
130  speed_pid_.set_setpoint(desired_speed);
131  desired_speed = speed_pid_.step(time_->dt(), state_->vel().norm());
132  }
133  vars_.output(speed_idx_, desired_speed);
134 
135  return true;
136 }
