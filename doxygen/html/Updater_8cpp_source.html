334 bool Updater::update_scale() {
335  for (auto &kv : actor_contacts_) {
336  // Get the desired scale amount, which is a factor of the current
337  // visualization scale (scale_) and the scale amount associated with
338  // the contact visual information
339  double desired_scale_amount = 1.0;
340  auto it = contact_visuals_.find(kv.first);
341  if (it != contact_visuals_.end()) {
342  desired_scale_amount = it->second->scale() * scale_;
343  }
344 
345  // Get the 4x4 matrix that defines the scale, rotation, and translation
346  vtkMatrix4x4 *m;
347  m = kv.second->actor->GetMatrix();
348 
349  // Compute the current scale factor for the object. The top-left 3x3
350  // part of the matrix contains the rotation and scale. The length of
351  // each row of the 3x3 matrix is the current scaling factor.
352  double sq_sum = 0;
353  for (int c = 0; c < 3; c++) {
354  sq_sum += pow(m->GetElement(0, c), 2);
355  }
356  double curr_scale_factor = sqrt(sq_sum);
357 
358  for (int r = 0; r < 3; r++) {
359  for (int c = 0; c < 3; c++) {
360  // For each element in the top-left 3x3 matrix, normalize the
361  // scaling factor to 1 and then apply the desired scaling
362  // amount.
363  double elem = m->GetElement(r, c);
364  elem = elem / curr_scale_factor * desired_scale_amount;
365  m->SetElement(r, c, elem);
366  }
367  }
368 
369  // update the label scale
370  kv.second->label->SetScale(label_scale_, label_scale_, label_scale_);
371  }
372  return true;
373 }

375 bool Updater::update_shapes() {
376  // Remove past frames shapes that have reached time-to-live, if they
377  // are not persistent
378  for (auto it = shapes_.begin(); it != shapes_.end();) {
379  bool remove = false;
380 
381  scrimmage_proto::Shape &s = std::get<0>(it->second);
382  if (not s.persistent()) {
383  switch (s.oneof_ttl_case()) {
384  case sp::Shape::kTtl:
385  s.set_ttl(s.ttl()-1);
386  if (s.ttl() <= 0) {
387  remove = true;
388  }
389  break;
390  case sp::Shape::kPersistDuration:
391  if (std::get<3>(it->second) + s.persist_duration() < frame_time_) {
392  remove = true;
393  }
394  break;
395  case sp::Shape::kPersistUntil:
396  if (s.persist_until() < frame_time_) {
397  remove = true;
398  }
399  break;
400  default:
401  break;
402  }
403  }
404 
405  if (remove) {
406  renderer_->RemoveActor(std::get<1>(it->second));
407  it = shapes_.erase(it);
408  } else {
409  ++it;
410  }
411  }
412  return true;
413 }

539 bool Updater::update_camera() {
540  // Free mode if contacts don't exist
541  if (actor_contacts_.empty()) {
542  set_view_mode(ViewMode::FREE);
543  return true;
544  }
545 
546  // Handle changing follow ids
547  if (inc_follow_) {
548  follow_id_++;
549  } else if (dec_follow_) {
550  follow_id_--;
551  }
552 
553  // Find min/max ids
554  // this is a map so the front will be the min and the back is the max
555  int min = actor_contacts_.begin()->first;
556  int max = actor_contacts_.rbegin()->first;
557 
558  auto it = actor_contacts_.find(follow_id_);
559  while (it == actor_contacts_.end()) {
560  if (follow_id_ > max) {
561  follow_id_ = actor_contacts_.begin()->second->contact.id().id();
562  } else if (follow_id_ < min) {
563  follow_id_ = actor_contacts_.rbegin()->second->contact.id().id();
564  } else if (inc_follow_) {
565  follow_id_++;
566  } else if (dec_follow_) {
567  follow_id_--;
568  } else {
569  // The ID might have been removed, increment to search for next
570  // available
571  follow_id_++;
572  }
573  it = actor_contacts_.find(follow_id_);
574  }
575  inc_follow_ = dec_follow_ = false;
576 
577  double x_pos = it->second->contact.state().position().x();
578  double y_pos = it->second->contact.state().position().y();
579  double z_pos = it->second->contact.state().position().z();
580 
581  double x_pos_fp = x_pos;
582  double y_pos_fp = y_pos;
583  double z_pos_fp = z_pos;
584 
585  double camera_pos[3] {0, 0, 0};
586  if (view_mode_ == ViewMode::FREE) {
587  if (reset_camera_) {
588  x_pos_fp = camera_reset_params_.focal_x;
589  y_pos_fp = camera_reset_params_.focal_y;
590  z_pos_fp = camera_reset_params_.focal_z;
591 
592  // there appears to be some type of gimbal lock in VTK
593  // so the y-position is slightly off of 0
594  camera_pos[0] = camera_reset_params_.pos_x;
595  camera_pos[1] = camera_reset_params_.pos_y;
596  camera_pos[2] = camera_reset_params_.pos_z;
597 
598  renderer_->GetActiveCamera()->SetViewUp(0, 1, 0);
599 
600  } else {
601  return true;
602  }
603  } else {
604  if (view_mode_ == ViewMode::OFFSET) {
605  camera_pos[0] = x_pos + 0.0;
606  camera_pos[1] = y_pos - 6.0;
607  camera_pos[2] = z_pos + 2.0;
608  } else if (view_mode_ == ViewMode::FOLLOW) {
609 
610  if (view_mode_prev_ == ViewMode::FOLLOW) {
611  double currentPos[3];
612  renderer_->GetActiveCamera()->GetPosition(currentPos);
613  double currentFp[3];
614  renderer_->GetActiveCamera()->GetFocalPoint(currentFp);
615  for (int i = 0; i < 3; i++)
616  follow_vec_[i] = currentPos[i]-currentFp[i];
617  follow_offset_ = follow_vec_.norm();
618  }
619 
620  Eigen::Vector3d rel_cam_pos = follow_vec_.normalized() * follow_offset_;
621  Eigen::Vector3d unit_vector = rel_cam_pos / rel_cam_pos.norm();
622 
623  sp::Quaternion sp_quat = it->second->contact.state().orientation();
624  sc::Quaternion quat(sp_quat.w(), sp_quat.x(), sp_quat.y(), sp_quat.z());
625 
626  Eigen::Vector3d pos = Eigen::Vector3d(x_pos, y_pos, z_pos) +
627  unit_vector * rel_cam_pos.norm();
628 
629  camera_pos[0] = pos[0];
630  camera_pos[1] = pos[1];
631  camera_pos[2] = pos[2];
632 
633  // Compute the camera's "up" vector
634  Eigen::Vector3d z_axis(0, 0, 1);
635  renderer_->GetActiveCamera()->SetViewUp(z_axis(0), z_axis(1), z_axis(2));
636 
637  } else if (view_mode_ == ViewMode::FPV) {
638  sp::Quaternion sp_quat = it->second->contact.state().orientation();
639  sc::Quaternion quat(sp_quat.w(), sp_quat.x(), sp_quat.y(), sp_quat.z());
640 
641  // Compute camera position
642  Eigen::Vector3d base_offset(1, 0, 0);
643  Eigen::Vector3d pos = Eigen::Vector3d(x_pos, y_pos, z_pos) +
644  quat * base_offset * 5;
645 
646  // Compute camera focal point
647  Eigen::Vector3d cam_dir = quat.rotate(base_offset.normalized());
648  Eigen::Vector3d fp = Eigen::Vector3d(x_pos, y_pos, z_pos) + cam_dir * 100;
649 
650  x_pos_fp = fp(0);
651  y_pos_fp = fp(1);
652  z_pos_fp = fp(2);
653 
654  camera_pos[0] = pos[0];
655  camera_pos[1] = pos[1];
656  camera_pos[2] = pos[2];
657 
658  // Compute the camera's "up" vector
659  Eigen::Vector3d z_axis(0, 0, 1);
660  Eigen::Vector3d up = quat * z_axis;
661  renderer_->GetActiveCamera()->SetViewUp(up(0), up(1), up(2));
662  }
663  }
664  view_mode_prev_ = view_mode_;

1698  vtkSmartPointer<vtkArrowSource>::New();

1701  double endPoint[3] = {a.head().x(), a.head().y(), a.head().z()};
1702

1704  double normalizedX[3] = {0.0, 0.0, 0.0};
1705  double normalizedY[3] = {0.0, 0.0, 0.0};
1706  double normalizedZ[3] = {0.0, 0.0, 0.0};
1707

1709  vtkMath::Subtract(endPoint, startPoint, normalizedX);
1710  double length = vtkMath::Norm(normalizedX);
1711  vtkMath::Normalize(normalizedX);
1712 
1713  // The Z axis is an arbitrary vector cross X
1714  double arbitrary[3] = {1.0, 2.0, 3.0};
1715  vtkMath::Cross(normalizedX, arbitrary, normalizedZ);
1716  vtkMath::Normalize(normalizedZ);
1717 
1718  // The Y axis is Z cross X
1719  vtkMath::Cross(normalizedZ, normalizedX, normalizedY);

1721  vtkSmartPointer<vtkMatrix4x4>::New();
1722 
1723  // Create the direction cosine matrix
1724  matrix->Identity();
1725  for (unsigned int i = 0; i < 3; i++) {
1726  matrix->SetElement(i, 0, normalizedX[i]);
1727  matrix->SetElement(i, 1, normalizedY[i]);
1728  matrix->SetElement(i, 2, normalizedZ[i]);
1729  }
1730 
1731  // Apply the transforms
1732  vtkSmartPointer<vtkTransform> transform =
1733  vtkSmartPointer<vtkTransform>::New();
1734  transform->Translate(startPoint);
1735  transform->Concatenate(matrix);
1736  transform->Scale(length, length, length);
1737 
1738  // Transform the polydata
1739  vtkSmartPointer<vtkTransformPolyDataFilter> transformPD =
1740  vtkSmartPointer<vtkTransformPolyDataFilter>::New();
1741  transformPD->SetTransform(transform);
1742  transformPD->SetInputConnection(arrowSource->GetOutputPort());

1748  mapper->SetInputConnection(transformPD->GetOutputPort());
1749 #endif
1750  actor->SetMapper(mapper);
1751  }
