76  if (follow_id_ < 0 || allow_prey_switching_) {

79  double min_dist = std::numeric_limits<double>::infinity();
80  for (auto it = contacts_->begin(); it != contacts_->end(); it++) {
81  // Skip if this contact isn't on team 1 (prey)

83  if (it->second.id().team_id() != prey_team_id_) continue;

87  if (dist < min_dist) {

90  min_dist = dist;

93  }

99  // Skip if this contact is on the same team
100  if (it->second.id().team_id() == parent_->id().team_id()) continue;
101 
102  // Calculate distance to entity
103  double dist = (it->second.state()->pos() - state_->pos()).norm();
104  // Publish capture message if within capture range

107  msg->data.set_source_id(parent_->id().id());
108  msg->data.set_target_id(it->second.id().id());
