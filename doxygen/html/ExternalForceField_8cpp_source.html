144 bool ExternalForceField::step_entity_interaction(std::list<sc::EntityPtr> &ents,
145  double t, double dt) {
146  if (force_type_ == Variable) {
147  if (time_->t() >= next_sample_time_) {
148  sample_force();
149  }
150  }
151 
152  auto msg_force = std::make_shared<sc::Message<Eigen::Vector3d>>();
153  msg_force->data = force_;
154  pub_->publish(msg_force);
155 
156  moment_(0) = roll_amp_ * sin(time_->t() * 2 * M_PI * 1.0 / roll_period_);
157  moment_(1) = pitch_amp_ * sin(time_->t() * 2 * M_PI * 1.0 / pitch_period_);
158  moment_(2) = yaw_amp_ * sin(time_->t() * 2 * M_PI * 1.0 / yaw_period_);
159 
160  for (auto &ent : ents) {
161  ent->motion()->set_external_force(force_);
162 
163  // Only apply moment force within z-boundary. Scale magnitude.
164  if (ent->state_truth()->pos()(2) >= moment_enable_min_z_ &&
165  ent->state_truth()->pos()(2) <= moment_enable_max_z_) {
166  double scale = (ent->state_truth()->pos()(2) - moment_enable_min_z_) /
167  (moment_enable_max_z_ - moment_enable_min_z_);
168  ent->motion()->set_external_moment(moment_ * scale);
169  }
170  }
171  return true;
