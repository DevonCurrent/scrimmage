251  fdm_pkt.zAccel = -specific_force_body(2);

254  fdm_pkt.yAccel = -state.linear_accel_body()(1);
255  fdm_pkt.zAccel = -state.linear_accel_body()(2);
256  }

259  fdm_pkt.roll = state.quat().roll();
260  fdm_pkt.pitch = -state.quat().pitch();

262

264  // world frame rotational velocities

279  fdm_pkt.yawRate = -state.ang_vel_body()(2);
280  }
