451 void FixedWing6DOF::model(const vector_t &x , vector_t &dxdt , double t) {
452  // Calculate velocity magnitude (handle zero velocity)
453  Eigen::Vector3d wind_B(wind_(0), -wind_(1), -wind_(2));
454  wind_B = quat_body_.rotate_reverse(wind_B);
455  Eigen::Vector3d vel_rel_wind(x_[U]+wind_B(0), x_[V]+wind_B(1), x_[W]+wind_B(2));
456  double V_tau = vel_rel_wind.norm();
457  if (std::abs(V_tau) < std::numeric_limits<double>::epsilon()) {
458  V_tau = 0.00001;
459  }
460 
461  // Calculate commonly used quantities
462  double beta = asin(clamp(vel_rel_wind(1)/V_tau, -1.0, 1.0)); // side slip
463  // double delta_Ve = 0.05*V_tau; // wind velocity across tail of aircraft (approximation)
464  // double VtauVe = pow((V_tau + delta_Ve)/V_tau, 2);
465  double pVtS = rho_ * pow(V_tau, 2) * S_ / 2.0;
466 
467 
468  // Calculate lift, drag, and side_force magnitudes
469  double CL = (C_L0_ + C_L_alpha_*alpha_ + C_LQ_*x_[Q] * c_ / (2*V_tau) +
470  C_L_alpha_dot_ * alpha_dot_ * c_ / (2*V_tau) +
471  C_L_delta_elevator_ * delta_elevator_);
472  double lift = CL * pVtS;
473 
474  double drag = (C_D0_ + C_D_alpha_*std::abs(alpha_) + CL*CL/(M_PI*AR_*e_) +
475  C_D_delta_elevator_*std::abs(delta_elevator_)) * pVtS;
476 
477 
478  double side_force = (C_Y_beta_*beta + C_Y_delta_rudder_*delta_rudder_) * pVtS;
479 
480  // Bring lift, drag, side_force magnitudes into body frame
481  Eigen::Vector3d F_aero(lift*sin(alpha_) - drag*cos(alpha_) - side_force*sin(beta),
482  side_force*cos(beta),
483  -lift*cos(alpha_) - drag*sin(alpha_));
484 
485  // Calculate force from weight in body frame:
486  Eigen::Vector3d gravity_vector(0, 0, +mass_*g_);
487  Eigen::Vector3d F_weight = quat_body_.rotate_reverse(gravity_vector);
488 
489  Eigen::Vector3d F_thrust(thrust_, 0, 0);
490  Eigen::Vector3d F_total = F_weight + F_thrust + F_aero;
491 
492  // simple ground contact model
493  Eigen::Vector3d F_ground_W(0, 0, 0);
494  Eigen::Vector3d F_ground(0, 0, 0);
495  if (x_[Zw] < 0 && use_ground_model_) {
496  double wn = 20;
497  double Kp = wn*wn*mass_;
498  double Kd = 2*wn;
499 
500  F_ground_W[2] = std::max(0.0, -Kp*(x_[Zw] - 0.0) - Kd*(x_[Ww] - 0.0));
501 
502  F_ground(0) = F_ground_W(0);
503  F_ground(1) = -F_ground_W(1);
504  F_ground(2) = -F_ground_W(2);
505  F_ground = quat_body_.rotate_reverse(F_ground);
506 
507  F_total += F_ground;
508  }
509 
510  if (POSTLAUNCH != launch_state_) {
511  F_total = force_ext_body_;
512  }
513 
514  // Calculate body frame linear velocities
515  dxdt[U] = x[V]*x[R] - x[W]*x[Q] + F_total(0) / mass_;
516  dxdt[V] = x[W]*x[P] - x[U]*x[R] + F_total(1) / mass_;
517  dxdt[W] = x[U]*x[Q] - x[V]*x[P] + F_total(2) / mass_;
518 
519  // Calculate moments;
520  Eigen::Vector3d Moments_thrust(0, 0, 0); // no moment from thrust
521  Eigen::Vector3d Moments_torque(0, 0, 0); // no moment from torque
522  Eigen::Vector3d Moments_gyro(0, 0, 0); // no moment from gyro effect
523 
524  // Calculate moments from aerodynamic forces
525  Eigen::Vector3d Moments_aero((C_L_beta_*beta + C_LP_*x_[P]*b_/(2*V_tau) + C_LR_*x_[R]*b_/(2*V_tau) + C_L_delta_aileron_*delta_aileron_ + C_L_delta_rudder_*delta_rudder_) * pVtS*b_,
526  (C_M0_ + C_M_alpha_*alpha_ + C_MQ_*x_[Q]*c_/(2*V_tau) + C_M_alpha_dot_*alpha_dot_*c_/(2*V_tau) + C_M_delta_elevator_*delta_elevator_) * pVtS*c_,
527  (C_N_beta_*beta + C_NP_*x_[P]*b_/(2*V_tau) + C_NR_*x_[R]*b_/(2*V_tau) + C_N_delta_aileron_*delta_aileron_ + C_N_delta_rudder_*delta_rudder_) * pVtS*b_);
528 
529  // Sum moments
530  Eigen::Vector3d Moments_total = Moments_aero + Moments_thrust +
531  Moments_torque + Moments_gyro;
532 
533  if (POSTLAUNCH != launch_state_) {
534  Moments_total == Eigen::Vector3d::Zero();
535  }
536 
537  // Calculate rotational velocites
538  Eigen::Vector3d pqr(x_[P], x_[Q], x_[R]);
539  Eigen::Vector3d pqr_dot = I_inv_ * (Moments_total - pqr.cross(I_*pqr));
540  dxdt[P] = pqr_dot(0);
541  dxdt[Q] = pqr_dot(1);
542  dxdt[R] = pqr_dot(2);
543 
544  // Compute quaternion derivatives
545  double lambda = 1 - (pow(x[q0], 2) + pow(x[q1], 2) + pow(x[q2], 2) + pow(x[q3], 2));
546  dxdt[q0] = -0.5 * (x[q1]*x[P] + x[q2]*x[Q] + x[q3]*x[R]) + lambda * x[q0];
547  dxdt[q1] = +0.5 * (x[q0]*x[P] + x[q2]*x[R] - x[q3]*x[Q]) + lambda * x[q1];
548  dxdt[q2] = +0.5 * (x[q0]*x[Q] + x[q3]*x[P] - x[q1]*x[R]) + lambda * x[q2];
549  dxdt[q3] = +0.5 * (x[q0]*x[R] + x[q1]*x[Q] - x[q2]*x[P]) + lambda * x[q3];
550 
551  // Normalize quaternion
552  sc::Quaternion quat(x[q0], x[q1], x[q2], x[q3]);
553  quat.normalize();
554 
555  // Integrate local velocities to compute local positions
556  Eigen::Vector3d vel_local(x[U], x[V], x[W]);
557  Eigen::Vector3d vel_world = quat.rotate(vel_local); // rot * vel_local;
558  dxdt[Xw] = vel_world(0);
559  dxdt[Yw] = -vel_world(1); // Due to rotated frame
560  dxdt[Zw] = -vel_world(2); // Due to rotated frame
561 
562  // Integrate local accelerations to compute global velocities
563  Eigen::Vector3d acc_local = F_total / mass_;
564  Eigen::Vector3d acc_world = quat.rotate(acc_local); // rot * acc_local;
565  dxdt[Uw] = acc_world(0);
566  dxdt[Vw] = -acc_world(1); // Due to rotated frame
567  dxdt[Ww] = -acc_world(2); // Due to rotated frame
568 }
