80 bool SphereNetwork::is_reachable(const scrimmage::EntityPluginPtr &pub_plugin,
81  const scrimmage::EntityPluginPtr &sub_plugin) {
82  // Never reachable if plugin's entity was destroyed
83  if (pub_plugin->parent() == nullptr || sub_plugin->parent() == nullptr) return false;
84  // If the publisher and subscriber have the same parent, it is reachable
85  if (pub_plugin->parent() == sub_plugin->parent()) return true;
86 
87  // Is the publisher to subscriber already in the reachable map?
88  int pub_id = pub_plugin->parent()->id().id();
89  int sub_id = sub_plugin->parent()->id().id();
90 
91  // Find the publisher in the mapping
92  auto it_pub = reachable_map_.find(pub_id);
93  if (it_pub != reachable_map_.end()) {
94  // See if the publisher has a reachability mapping for the subscriber
95  auto it_sub = it_pub->second.find(sub_id);
96  if (it_sub != it_pub->second.end()) {
97  return it_sub->second;
98  }
99  }
100 
101  // No reachability mapping exists. Find all entity IDs that are reachable.
102  std::vector<sc::ID> neigh;
103  rtree_->neighbors_in_range(pub_plugin->parent()->state_truth()->pos(),
104  neigh, range_);
105 
106  // Add the entity IDs to the publisher's reachability set
107  // Look for the subscriber ID
108  reachable_map_[pub_id][sub_id] = false; // Default to not reachable
109  reachable_map_[sub_id][pub_id] = false; // Default to not reachable
110  for (sc::ID id : neigh) {
111  auto ent_neighbor = id_to_ent_map_->find(id.id());
112  if (ent_neighbor == id_to_ent_map_->end()) {
113  std::cout << "Warning: Sphere Network entity id doesn't exists: "
114  << id.id() << std::endl;
115  continue;
116  }
117  if (not filter_comms_plane_) {
118  reachable_map_[pub_id][id.id()] = true;
119  reachable_map_[id.id()][pub_id] = true;
120  } else if (within_planar_boundary(
121  pub_plugin->parent()->state_truth()->pos()[2],
122  ent_neighbor->second->state_truth()->pos()[2])) {
123  reachable_map_[pub_id][id.id()] = true;
124  reachable_map_[id.id()][pub_id] = true;
125  }
126  }
127  return reachable_map_[pub_id][sub_id];
128 }

136 bool SphereNetwork::within_planar_boundary(double z1, double z2) {
