115 double PID::step(const double &dt, const double &measurement) {
116  error_ = setpoint_ - measurement;
117  if (is_angle_) {
118  error_ = Angles::angle_pi(error_);
119  }
120 
121  if (std::abs(error_) > integral_band_) {
122  integral_ = 0;
123  } else {
124  integral_ += error_ * dt;
125  }
126 
127  derivative_ = (error_ - prev_error_) / std::max(1.0e-9, dt);
128  double u = kp_*error_ + ki_*integral_ + kd_*derivative_;
129  prev_error_ = error_;
130 
131  return enable_output_limits_ ? clamp(u, output_min_, output_max_) : u;
