205 bool CSV::read_csv_from_string(const std::string &csv_str, const bool& contains_header) {
206  table_.clear();
207  column_headers_.clear();
208 
209  std::vector<std::string> line_tokens;
210  boost::split(line_tokens, csv_str, boost::is_any_of("\n"));
211  int row_num = 0;
212  for (unsigned int line_num = 0; line_num < line_tokens.size(); line_num++) {
213  // Strip "whitespace" characters.
214  std::string line = remove_whitespace(line_tokens[line_num]);
215 
216  if (line == "") continue; // Ignore lines that are empty
217 
218  if (line_num == 0 && contains_header) {
219  std::list<std::string> headers = get_csv_line_elements(line);
220  int i = 0;
221  for (auto &str : headers) {
222  column_headers_[str] = i;
223  i++;
224  }
225  } else {
226  std::vector<std::string> tokens;
227  boost::split(tokens, line, boost::is_any_of(","));
228  for (unsigned int i = 0; i < tokens.size(); i++) {
229  table_[row_num][i] = std::stod(tokens[i]);
230  }
231 
232  // If this is the first line and the file doesn't contain a header,
233  // populate the column headers with indices based on the number of
234  // comma separated values
235  if (row_num == 0 && not contains_header) {
236  for (unsigned int i = 0; i < tokens.size(); i++) {
237  column_headers_[std::to_string(i)] = i;
238  }
239  }
240 
241  // Print a warning if the number of comma separated values doesn't
242  // match the number of columns
243  if (column_headers_.size() != tokens.size()) {
244  cout << "Warning the number of values (" << tokens.size()
245  << ") on line number " << row_num
246  << " doesn't match the number of column headers: "
247  << column_headers_.size() << endl;
248  }
249  row_num++;
250  }
251  }
252  return true;
253 }
